    55→def compute_atr_pct(df, period: int = 14) -> np.ndarray:
    56→    """
    57→    Compute ATR (Average True Range) as a percentage of close price.
    58→
    59→    ATR measures volatility — higher ATR = more volatile = wider stops needed.
    60→    Expressing ATR as % of price makes it comparable across price levels.
    61→
    62→    Args:
    63→        df: OHLCV DataFrame
    64→        period: ATR lookback period (default 14)
    65→
    66→    Returns:
    67→        numpy array of ATR % values, same length as df
    68→    """
    69→    high = df['high'].values.astype(float)
    70→    low = df['low'].values.astype(float)
    71→    close = df['close'].values.astype(float)
    72→
    73→    # True Range = max(H-L, |H-prev_C|, |L-prev_C|)
    74→    prev_close = np.roll(close, 1)
    75→    prev_close[0] = close[0]
    76→    tr = np.maximum(
    77→        high - low,
    78→        np.maximum(
    79→            np.abs(high - prev_close),
    80→            np.abs(low - prev_close)
    81→        )
    82→    )
    83→
    84→    # Wilder's smoothing (exponential moving average)
    85→    atr = np.zeros_like(tr)
    86→    if len(tr) >= period:
    87→        atr[period - 1] = np.mean(tr[:period])
    88→        for i in range(period, len(tr)):
    89→            atr[i] = (atr[i - 1] * (period - 1) + tr[i]) / period
    90→
    91→    # Convert to percentage of close price
    92→    atr_pct = np.where(close > 0, atr / close * 100, 0)
    93→    return atr_pct

    96→def _record_observation(
    97→    trie: "PPMTTrie",
    98→    trade: PaperTrade,
    99→    exit_sym_idx: int,
   100→    next_symbol: Optional[str] = None,
   101→    fuzzy_matcher: Optional["FuzzyMatcher"] = None,
   102→) -> dict:
   103→    """
   104→    Living Trie: Record a trade's outcome back into the Trie.
   105→
   106→    This is the core of the "Living Trie" concept — the Trie learns from
   107→    its own trading results. Every closed trade becomes a new observation
   108→    that updates the node's metadata, creating a feedback loop:
   109→
   110→      Trie predicts → Trade executes → Outcome observed → Trie updated
   111→
   112→    v0.6.6 FIX: Read/Write Path Alignment
   113→    Previously, the READ path used FuzzyMatcher (allowing 1-edit matches)
   114→    but the WRITE path used trie.search() (exact match only). When a trade
   115→    was entered via fuzzy match, _record_observation couldn't find the exact
   116→    pattern and created a NEW branch — causing node proliferation.
   117→
   118→    Now, if fuzzy_matcher is provided:
   119→    - Path A: Use best_match() to find the closest existing node instead
   120→      of creating a new branch for patterns that are 1-edit away.
   121→    - Path B: Use check_continuation() to find fuzzy-close children
   122→      instead of always creating new children for unseen continuations.
   123→
   124→    Two things happen:
   125→    1. The entry node's metadata is updated with the actual outcome
   126→       (win/loss, actual move, duration)
   127→    2. If a pattern break occurred with a new symbol, that symbol is
   128→       added as a new child node — the Trie literally grows
   129→
   130→    Args:
   131→        trie: The PPMT Trie to update
   132→        trade: The closed PaperTrade with outcome data
   133→        exit_sym_idx: SAX symbol index at trade exit
   134→        next_symbol: The SAX symbol that followed (especially important
   135→                     for pattern breaks — this is the NEW symbol)
   136→        fuzzy_matcher: Optional FuzzyMatcher for read/write alignment.
   137→            When provided, uses fuzzy matching to find the closest
   138→            existing node instead of creating new branches.
   139→
   140→    Returns:
   141→        Dict with 'observations' and 'new_nodes' counts
   142→    """
   143→    from ppmt.core.metadata import BlockLifecycleMetadata
   144→
   145→    observations = 0
   146→    new_nodes = 0
   147→
   148→    if not trade.matched_pattern:
   149→        return {"observations": 0, "new_nodes": 0}
   150→
   151→    # 1. Find the Trie node for the matched entry pattern
   152→    #    v0.6.6: Try exact match first, then fuzzy, then prefix fallback.
   153→    #    This aligns the WRITE path with the READ path — if a trade was
   154→    #    entered via fuzzy match, we find the same node to write back to.
   155→    node = trie.search(trade.matched_pattern)
   156→    matched_via_fuzzy = False
   157→    matched_pattern = trade.matched_pattern  # Track which pattern we matched
   158→
   159→    if node is None and fuzzy_matcher is not None:
   160→        # v0.6.6 FIX: Use FuzzyMatcher to find closest existing node.
   161→        # This prevents creating duplicate branches for patterns that
   162→        # are 1-edit away from existing patterns. The trade was likely
   163→        # entered via a fuzzy match, so we should write back to the
   164→        # same node that generated the entry signal.
   165→        fuzzy_result = fuzzy_matcher.best_match(trie, trade.matched_pattern)
   166→        if fuzzy_result.matched and fuzzy_result.node is not None:
   167→            node = fuzzy_result.node
   168→            matched_via_fuzzy = True
   169→            matched_pattern = fuzzy_result.symbols
   170→
   171→    if node is None:
   172→        # Try shorter prefixes (the prediction may have used a prefix match)
   173→        for prefix_len in range(len(trade.matched_pattern) - 1, 0, -1):
   174→            node = trie.search(trade.matched_pattern[:prefix_len])
   175→            if node is not None:
   176→                matched_pattern = trade.matched_pattern[:prefix_len]
   177→                break
   178→
   179→    if node is None:
   180→        # Pattern not in Trie at all — create the entry as a new pattern
   181→        # V4.4: Pass regime/regime_confidence so new nodes inherit regime context.
   182→        # Previously these were missing, causing newly-created Living Trie nodes
   183→        # to have empty regime info, which broke regime-aware confidence scoring.
   184→        #
   185→        # v0.6.6 NOTE: This should be RARE with fuzzy_matcher enabled.
   186→        # Only truly novel patterns (no fuzzy match) create new branches.
   187→        trie.insert_with_observations(
   188→            symbols=trade.matched_pattern,
   189→            move_pct=trade.actual_move_pct,
   190→            drawdown_pct=min(trade.actual_move_pct, 0) if trade.actual_move_pct < 0 else 0,
   191→            favorable_pct=max(trade.actual_move_pct, 0) if trade.actual_move_pct > 0 else 0,
   192→            duration=max(1, exit_sym_idx - trade.entry_sym_idx) if trade.entry_sym_idx > 0 else 1,
   193→            # v0.40.23 (P7-FaseC): outcome-SL/TP based won (see update above).
   194→            won=(trade.exit_reason == "take_profit"),
   195→            next_symbol=next_symbol,
   196→            regime=trade.regime if trade.regime else None,
   197→            regime_confidence=trade.regime_confidence if trade.regime_confidence > 0 else None,
   198→        )
   199→        new_nodes += 1
   200→        observations += 1
   201→        trie.trading_observations += 1
   202→        trade.trie_updated = True
   203→        return {"observations": observations, "new_nodes": new_nodes}
   204→
   205→    # 2. Update the node's metadata with the actual trade outcome
   206→    duration = max(1, exit_sym_idx - trade.entry_sym_idx) if trade.entry_sym_idx > 0 else 1
   207→    # v0.40.23 (P7-FaseC): `won` is now outcome-SL/TP based. The trade already
   208→    # knows its exit_reason, so we just map it: take_profit → win, anything else
   209→    # (stop_loss / trailing_stop / pattern_break / end_of_data / catastrophic_stop)
   210→    # → loss. This replaces the legacy `trade.pnl_pct > 0` check which conflated
   211→    # "barely positive exit" with "TP touched first" — the two are very different:
   212→    # a trade that grazed SL before recovering to +0.01% PnL is NOT a clean win.
   213→    won = (trade.exit_reason == "take_profit")
   214→
   215→    # Drawdown: SL distance is max drawdown estimate
   216→    if trade.sl_price > 0 and trade.entry_price > 0:
   217→        if trade.direction == "LONG":
   218→            max_dd_pct = -(trade.entry_price - trade.sl_price) / trade.entry_price * 100
   219→        else:
   220→            max_dd_pct = -(trade.sl_price - trade.entry_price) / trade.entry_price * 100
   221→    else:
   222→        max_dd_pct = min(trade.actual_move_pct, 0)
   223→
   224→    # Favorable: TP distance is max favorable estimate
   225→    if trade.tp_price > 0 and trade.entry_price > 0:
   226→        if trade.direction == "LONG":
   227→            max_fav_pct = (trade.tp_price - trade.entry_price) / trade.entry_price * 100
   228→        else:
   229→            max_fav_pct = (trade.entry_price - trade.tp_price) / trade.entry_price * 100
   230→    else:
   231→        max_fav_pct = max(trade.actual_move_pct, 0)
   232→
   233→    # Update entry node metadata
   234→    node.metadata.update_from_observation(
   235→        move_pct=trade.actual_move_pct,
   236→        drawdown_pct=max_dd_pct,
   237→        favorable_pct=max_fav_pct,
   238→        duration=duration,
   239→        won=won,
   240→        next_symbol=next_symbol,
   241→        regime=trade.regime if trade.regime else None,
   242→        regime_confidence=trade.regime_confidence if trade.regime_confidence > 0 else None,
   243→    )
   244→    observations += 1
   245→    trie.trading_observations += 1
   246→
   247→    # 3. If next_symbol is NOT already a child, add it — the Trie GROWS
   248→    #    v0.6.6 FIX: Check fuzzy continuation before creating new child.
   249→    #    If a fuzzy-close symbol already exists as a child, write the
   250→    #    observation to THAT child instead of creating a new branch.
   251→    if next_symbol:
   252→        if node.has_child(next_symbol):
   253→            # Exact child exists — update it
   254→            child_node = node.get_child(next_symbol)
   255→            child_node.metadata.update_from_observation(
   256→                move_pct=trade.actual_move_pct * 0.5,
   257→                drawdown_pct=max_dd_pct,
   258→                favorable_pct=max_fav_pct,
   259→                duration=duration,
   260→                won=won,
   261→                next_symbol=None,
   262→                regime=trade.regime if trade.regime else None,
   263→                regime_confidence=trade.regime_confidence if trade.regime_confidence > 0 else None,
   264→            )
   265→        elif fuzzy_matcher is not None:
   266→            # v0.6.6: Check if a fuzzy-close continuation exists.
   267→            # If the next symbol is similar to an existing child,
   268→            # write to that child instead of creating a new one.
   269→            cont_result = fuzzy_matcher.check_continuation(
   270→                trie, matched_pattern, next_symbol
   271→            )
   272→            if cont_result.matched and cont_result.node is not None:
   273→                # Fuzzy continuation found — write to existing child
   274→                cont_result.node.metadata.update_from_observation(
   275→                    move_pct=trade.actual_move_pct * 0.5,
   276→                    drawdown_pct=max_dd_pct,
   277→                    favorable_pct=max_fav_pct,
   278→                    duration=duration,
   279→                    won=won,
   280→                    next_symbol=None,
   281→                    regime=trade.regime if trade.regime else None,
   282→                    regime_confidence=trade.regime_confidence if trade.regime_confidence > 0 else None,
   283→                )
   284→            else:
   285→                # No fuzzy continuation — create new child (genuinely novel)
   286→                extended_pattern = matched_pattern + [next_symbol]
   287→                child_node = trie.insert(extended_pattern)
   288→                child_node.metadata.update_from_observation(
   289→                    move_pct=trade.actual_move_pct * 0.5,
   290→                    drawdown_pct=max_dd_pct,
   291→                    favorable_pct=max_fav_pct,
   292→                    duration=duration,
   293→                    won=won,
   294→                    next_symbol=None,
   295→                    regime=trade.regime if trade.regime else None,
   296→                    regime_confidence=trade.regime_confidence if trade.regime_confidence > 0 else None,
   297→                )
   298→                new_nodes += 1
   299→        else:
   300→            # No fuzzy matcher — original behavior (always create new child)
   301→            extended_pattern = matched_pattern + [next_symbol]
   302→            child_node = trie.insert(extended_pattern)
   303→            child_node.metadata.update_from_observation(
   304→                move_pct=trade.actual_move_pct * 0.5,
   305→                drawdown_pct=max_dd_pct,
   306→                favorable_pct=max_fav_pct,
   307→                duration=duration,
   308→                won=won,
   309→                next_symbol=None,
   310→                regime=trade.regime if trade.regime else None,
   311→                regime_confidence=trade.regime_confidence if trade.regime_confidence > 0 else None,
   312→            )
   313→            new_nodes += 1
   314→
   315→    trade.trie_updated = True
   316→    return {"observations": observations, "new_nodes": new_nodes}

   657→class PaperTrader:
   658→    """
   659→    Paper Trading Engine using PPMT predictions.
   660→
   661→    Steps through historical candles, generating predictions and
   662→    executing simulated trades with adaptive position sizing.
   663→
   664→    This is the key validation tool: run paper trading on historical
   665→    data to verify PPMT predictions before risking real capital.
   666→    """
   667→
   668→    def __init__(self, config: Optional[PaperTraderConfig] = None):
   669→        self.config = config or PaperTraderConfig()
   670→        self.risk_config = RiskConfig(
   671→            base_position_size_pct=0.01,  # 1% risk per trade (conservative while tuning)
   672→            max_position_size_pct=0.04,
   673→            min_position_size_pct=0.005,
   674→            min_risk_reward=1.0,         # v0.2.9: lowered from 1.5 to allow SHORT R:R=1.5
   675→            max_daily_loss_pct=0.10,      # 10% daily loss limit
   676→            # v0.40.6 FIX-8 (CAPA 5 audit): lowered from 0.80 to 0.20.
   677→            # 0.80 allowed losing 80% of capital before blocking new trades
   678→            # (80 trades @ 1% loss each). 0.20 is still generous for paper
   679→            # trading but prevents catastrophic drawdowns from going unblocked.
   680→            # For real trading, recommend 0.15 (15%).
   681→            max_drawdown_pct=0.20,        # 20% max drawdown (was 0.80)
   682→            min_quality_score=0.0,        # Checked in paper_trader, not RiskManager
   683→            # min_confidence is checked in PaperTrader.run(), not RiskManager
   684→        )
   685→
   686→    def run(self) -> PaperTraderResult:
   687→        """
   688→        Run paper trading simulation on stored historical data.
   689→
   690→        v0.6.4: TokenProfile integration.
   691→        The paper trader now auto-configures from TokenProfile:
   692→        - SAX alpha/window from timeframe-adaptive mapping (1h→3, 5m→4, 1m→5)
   693→        - catastrophic_loss_pct from asset class (blue_chip:8%, meme:15%, etc.)
   694→        - short_allowed / short_confidence_multiplier from asset class
   695→        - fuzzy_threshold from asset class
   696→        This replaces manual per-token tuning with data-driven
   697→        auto-configuration validated across 22 token-TF combos with
   698→        6+ months of real Binance data each.
   699→
   700→        v0.3.0: Revert to v0.2.8 SL/TP behavior for maximum P&L.
   701→        v0.2.10's "improvements" reduced P&L from +1578% to +371%:
   702→        - Catastrophic protection (5%) cut trades that would reach TP
   703→        - Trailing stop activated too early (50% of TP distance)
   704→        - LONG SL floor of 2.0% was too wide vs v0.2.8's 1.5%
   705→        - min_confidence 0.12 filtered valid entries
   706→
   707→        v0.3.0 changes (back to v0.2.8 baseline):
   708→        - SL/TP checked at SAX window boundaries ONLY (no intra-window)
   709→        - No catastrophic protection (disabled by default)
   710→        - LONG SL: max(ATR*1.5, 1.5%), cap 5% (v0.2.8 values)
   711→        - SHORT SL: max(ATR*2.0, 2.0%), cap 7% (v0.2.8 values)
   712→        - Trailing stop activates at 75% of TP distance (not 50%)
   713→        - min_confidence = 0.10 (v0.2.8 value)
   714→        - min_quality_score = 0.0 (removed, Living Trie handles this)
   715→
   716→        Kept from v0.2.9/v0.2.10:
   717→        - Pattern break grace period (2 consecutive)
   718→        - Re-entry cooldown = 1 symbol step
   719→        - Living Trie (ON by default)
   720→        - Direction-specific SL/TP (SHORT gets wider stops)
   721→        """
   722→        cfg = self.config
   723→        storage = PPMTStorage()
   724→
   725→        # Load data
   726→        df = storage.load_ohlcv(cfg.symbol, cfg.timeframe)
   727→        if df.empty:
   728→            console.print(f"[red]No data for {cfg.symbol}. Run 'ppmt ingest' first.[/red]")
   729→            return PaperTraderResult(symbol=cfg.symbol, timeframe=cfg.timeframe)
   730→
   731→        # Classify asset
   732→        classifier = AssetClassifier()
   733→        info = classifier.classify(cfg.symbol)
   734→
   735→        # ================================================================
   736→        # v0.6.4: TokenProfile integration
   737→        # Auto-configure SAX parameters, risk params, and gating from
   738→        # the TokenProfile based on asset class + timeframe.
   739→        # ================================================================
   740→        token_profile = None
   741→        if cfg.use_token_profile:
   742→            # v0.11.0: Try to load a previously saved calibrated profile
   743→            # This skips the expensive TradingCalibrationEngine grid search
   744→            # when a profile was already calibrated in a previous run.
   745→            saved_profile_dict = storage.load_token_profile(cfg.symbol, cfg.timeframe)
   746→
   747→            if saved_profile_dict is not None:
   748→                try:
   749→                    token_profile = TokenProfile.from_dict(saved_profile_dict)
   750→                    console.print(f"[bold green]TokenProfile restored from storage:[/bold green] "
   751→                                  f"{token_profile.asset_class} @ {cfg.timeframe} → "
   752→                                  f"alpha={token_profile.sax_alphabet_size}, "
   753→                                  f"window={token_profile.sax_window_size}, "
   754→                                  f"calibrated={token_profile.calibration_date}, "
   755→                                  f"changes={token_profile.profile_changes}")
   756→                except Exception as e:
   757→                    console.print(f"[yellow]Saved profile corrupt ({e}) — creating fresh[/yellow]")
   758→                    token_profile = None
   759→
   760→            if token_profile is None:
   761→                token_profile = TokenProfile.from_timeframe(
   762→                    symbol=cfg.symbol,
   763→                    asset_class=info.asset_class,
   764→                    timeframe=cfg.timeframe,
   765→                )
   766→
   767→            # Override SAX params from profile (unless explicitly set)
   768→            if cfg.sax_alphabet_size == 0:
   769→                cfg.sax_alphabet_size = token_profile.sax_alphabet_size
   770→            if cfg.sax_window_size == 0:
   771→                cfg.sax_window_size = token_profile.sax_window_size
   772→            # Override catastrophic_loss from profile (unless explicitly set)
   773→            if cfg.catastrophic_loss_pct == 0.0:
   774→                cfg.catastrophic_loss_pct = token_profile.catastrophic_loss_pct * 100.0
   775→
   776→            console.print(f"[bold green]TokenProfile loaded:[/bold green] "
   777→                          f"{info.asset_class} @ {cfg.timeframe} → "
   778→                          f"alpha={token_profile.sax_alphabet_size}, "
   779→                          f"window={token_profile.sax_window_size}, "
   780→                          f"cat_loss={token_profile.catastrophic_loss_pct:.0%}, "
   781→                          f"short_allowed={token_profile.short_allowed}, "
   782→                          f"fuzzy={token_profile.fuzzy_threshold:.2f}")
   783→
   784→            # ================================================================
   785→            # v0.6.8: Auto-calibration with TradingCalibrationEngine
   786→            # Run a mini-backtest grid search on the available data to
   787→            # discover the best α/W for THIS specific token, overriding
   788→            # the generic timeframe defaults. This fixes the structural
   789→            # bias of the old CalibrationEngine (always α=3/W=5) by
   790→            # selecting based on actual trading PnL.
   791→            #
   792→            # v0.11.0: Skip calibration if a profile was already restored
   793→            # from storage (calibration_date is set). This saves significant
   794→            # startup time on subsequent runs. The recalibration_interval
   795→            # mechanism handles periodic re-verification during trading.
   796→            # ================================================================
   797→            profile_already_calibrated = (
   798→                token_profile.calibration_date != ""
   799→                and token_profile.calibration_metric > 0
   800→            )
   801→            if profile_already_calibrated:
   802→                console.print(f"[green]Profile already calibrated on "
   803→                              f"{token_profile.calibration_date} "
   804→                              f"(metric={token_profile.calibration_metric:.4f}, "
   805→                              f"changes={token_profile.profile_changes}) — "
   806→                              f"skipping calibration[/green]")
   807→            elif cfg.auto_calibrate and len(df) >= 1000:
   808→                console.print(f"[bold cyan]Auto-calibrating α/W...[/bold cyan] "
   809→                              f"({len(df)} candles, this may take a moment)")
   810→                try:
   811→                    calibrator = TradingCalibrationEngine(
   812→                        train_ratio=0.70,
   813→                        pattern_length=cfg.pattern_length,
   814→                        timeframe=cfg.timeframe,
   815→                    )
   816→                    cal_profile, cal_results = calibrator.calibrate(
   817→                        df, symbol=cfg.symbol, verbose=False
   818→                    )
   819→                    cal_alpha = cal_profile.sax_alphabet_size
   820→                    cal_window = cal_profile.sax_window_size
   821→
   822→                    # Find the best result for display
   823→                    cal_best = [r for r in cal_results
   824→                                if r.alphabet_size == cal_alpha
   825→                                and r.window_size == cal_window][0]
   826→
   827→                    # Only override if calibration found a different α/W
   828→                    if cal_alpha != cfg.sax_alphabet_size or cal_window != cfg.sax_window_size:
   829→                        old_alpha, old_window = cfg.sax_alphabet_size, cfg.sax_window_size
   830→                        cfg.sax_alphabet_size = cal_alpha
   831→                        cfg.sax_window_size = cal_window
   832→
   833→                        # Update the token profile with calibrated values
   834→                        grid = {
   835→                            f"a{r.alphabet_size}_w{r.window_size}": {
   836→                                "trading_metric": round(r.trading_metric, 4),
   837→                                "total_pnl_pct": round(r.total_pnl_pct, 2),
   838→                                "win_rate": round(r.win_rate, 4),
   839→                                "total_trades": r.total_trades,
   840→                            }
   841→                            for r in cal_results
   842→                        }
   843→                        token_profile.update_from_calibration(
   844→                            best_alpha=cal_alpha,
   845→                            best_window=cal_window,
   846→                            metric=cal_best.trading_metric,
   847→                            grid=grid,
   848→                            n_samples=len(df),
   849→                        )
   850→
   851→                        console.print(f"[bold green]Calibrated:[/bold green] "
   852→                                      f"α={old_alpha}→{cal_alpha}, W={old_window}→{cal_window} "
   853→                                      f"(PnL={cal_best.total_pnl_pct:+.1f}%, "
   854→                                      f"WR={cal_best.win_rate:.1%}, "
   855→                                      f"Trades={cal_best.total_trades}, "
   856→                                      f"SL/TP={calibrator.sl_pct}%/{calibrator.tp_pct}%)")
   857→                    else:
   858→                        console.print(f"[green]Calibration confirms default: "
   859→                                      f"α={cal_alpha}/W={cal_window} "
   860→                                      f"(PnL={cal_best.total_pnl_pct:+.1f}%, "
   861→                                      f"WR={cal_best.win_rate:.1%})[/green]")
   862→                except Exception as e:
   863→                    console.print(f"[yellow]Calibration failed: {e} — "
   864→                                  f"using default α/W[/yellow]")
   865→
   866→            # v0.11.0: Save calibrated profile to storage for next run
   867→            if token_profile is not None and token_profile.calibration_date:
   868→                try:
   869→                    storage.save_token_profile(cfg.symbol, cfg.timeframe, token_profile.to_dict())
   870→                    console.print(f"[dim]  TokenProfile saved to storage "
   871→                                  f"(α={token_profile.sax_alphabet_size}, "
   872→                                  f"W={token_profile.sax_window_size})[/dim]")
   873→                except Exception as e:
   874→                    console.print(f"[dim]  Failed to save profile: {e}[/dim]")
   875→        else:
   876→            # Fallback: use timeframe alpha defaults even without full profile
   877→            if cfg.sax_alphabet_size == 0 or cfg.sax_window_size == 0:
   878→                tf_defaults = TIMEFRAME_ALPHA_DEFAULTS.get(
   879→                    cfg.timeframe, TIMEFRAME_ALPHA_DEFAULTS["1h"]
   880→                )
   881→                if cfg.sax_alphabet_size == 0:
   882→                    cfg.sax_alphabet_size = tf_defaults["sax_alphabet_size"]
   883→                if cfg.sax_window_size == 0:
   884→                    cfg.sax_window_size = tf_defaults["sax_window_size"]
   885→
   886→            if cfg.catastrophic_loss_pct == 0.0:
   887→                cfg.catastrophic_loss_pct = 8.0  # safe default
   888→
   889→        # v0.40.6 FIX-9 (CAPA 5 audit): Guarantee catastrophic protection
   890→        # is at least 3.0% when not explicitly disabled. Previously when
   891→        # use_token_profile=False and catastrophic_loss_pct defaulted to 0.0,
   892→        # the system had NO intra-window protection — a flash crash within
   893→        # a SAX window (5 candles of 1m = 5 min) would not be detected
   894→        # until the window closed. For TF 1m/5m with maximum entry frequency,
   895→        # this is critical. 3.0% is conservative — production should tune
   896→        # per asset class via TokenProfile.
   897→        if cfg.catastrophic_loss_pct < 3.0:
   898→            cfg.catastrophic_loss_pct = 3.0
   899→
   900→        # Try to load existing Tries, or build new ones
   901→        # v0.10.0: Load all 4 levels for GAP-1 4-level matching
   902→        # v0.40.4 FIX-1D: pass asset_class so N1/N2 load from cross-asset pools.
   903→        all_tries = storage.load_all_tries(cfg.symbol, asset_class=info.asset_class)
   904→        trie_n1 = all_tries["n1"]
   905→        trie_n2 = all_tries["n2"]
   906→        trie_n3 = all_tries["n3"]
   907→        trie_n4 = all_tries["n4"]
   908→
   909→        initial_pattern_count = 0
   910→        has_multi_level = (
   911→            cfg.use_multi_level
   912→            and trie_n1 is not None
   913→            and trie_n2 is not None
   914→            and trie_n4 is not None
   915→        )
   916→
   917→        # v0.6.4: Get fuzzy_threshold from TokenProfile if available
   918→        fuzzy_threshold = 0.80  # safe default
   919→        if token_profile is not None:
   920→            fuzzy_threshold = token_profile.fuzzy_threshold
   921→
   922→        if trie_n3 is None:
   923→            console.print(f"[yellow]No Trie for {cfg.symbol}. Building from data...[/yellow]")
   924→            engine = PPMT(
   925→                symbol=cfg.symbol,
   926→                asset_class=info.asset_class,
   927→                sax_alphabet_size=cfg.sax_alphabet_size,
   928→                sax_window_size=cfg.sax_window_size,
   929→                sax_strategy=cfg.sax_strategy,
   930→                fuzzy_threshold=fuzzy_threshold,
   931→                weight_profile=info.weight_profile,
   932→            )
   933→            # FIX-1D: contribute to universal N1 + class N2 pools
