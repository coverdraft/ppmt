    49‚Üí    49‚Üídef parse_mexc_orders(xlsx_path: Path) -> pd.DataFrame:
    50‚Üí    50‚Üí    """Parse MEXC futures order history."""
    51‚Üí    51‚Üí    LOG.info("Loading MEXC orders from %s", xlsx_path.name)
    52‚Üí    52‚Üí    df = pd.read_excel(xlsx_path)
    53‚Üí    53‚Üí    LOG.info("Loaded %d orders", len(df))
    54‚Üí    54‚Üí    
    55‚Üí    55‚Üí    # Parse timestamps (MEXC gives UTC+2, convert to UTC)
    56‚Üí    56‚Üí    df["_ts"] = pd.to_datetime(df["Tiempo(UTC+02:00)"].astype(str)) - pd.Timedelta(hours=2)
    57‚Üí    57‚Üí    df["_ts"] = df["_ts"].dt.tz_localize("UTC")
    58‚Üí    58‚Üí    
    59‚Üí    59‚Üí    # Parse symbol: "RIVERUSDT" ‚Üí "RIVER"
    60‚Üí    60‚Üí    df["_symbol"] = df["Par de Trading de Futuros"].str.strip().str.upper().str.replace("USDT", "", regex=False).str.strip()
    61‚Üí    61‚Üí    
    62‚Üí    62‚Üí    # Parse direction: MEXC semantics (VERIFIED from actual order flow)
    63‚Üí    63‚Üí    #   buy long   = OPEN LONG    (PNL always 0)
    64‚Üí    64‚Üí    #   sell long  = CLOSE LONG   (PNL = trade result)
    65‚Üí    65‚Üí    #   buy short  = OPEN SHORT   (PNL always 0)  ‚Üê was WRONG before!
    66‚Üí    66‚Üí    #   sell short = CLOSE SHORT  (PNL = trade result)  ‚Üê was WRONG before!
    67‚Üí    67‚Üí    #
    68‚Üí    68‚Üí    # Evidence: RIVERUSDT short trade: buy short @ 4.327, 4.314 ‚Üí sell short @ 4.294 (PnL +1.97)
    69‚Üí    69‚Üí    # The "buy short" orders open the position, "sell short" closes it with PnL.
    70‚Üí    70‚Üí    df["_action"] = df["Direcci√≥n"].str.strip().str.lower().map({
    71‚Üí    71‚Üí        "buy long": "open_long",
    72‚Üí    72‚Üí        "sell long": "close_long", 
    73‚Üí    73‚Üí        "buy short": "open_short",
    74‚Üí    74‚Üí        "sell short": "close_short",
    75‚Üí    75‚Üí    })
    76‚Üí    76‚Üí    
    77‚Üí    77‚Üí    # Numeric columns
    78‚Üí    78‚Üí    df["_price"] = pd.to_numeric(df["Precio promedio completo"], errors="coerce")
    79‚Üí    79‚Üí    df["_pnl"] = pd.to_numeric(df["PNL de Cierre"], errors="coerce").fillna(0)
    80‚Üí    80‚Üí    df["_fee"] = pd.to_numeric(df["Comisi√≥n de Trading"], errors="coerce").fillna(0).abs()
    81‚Üí    81‚Üí    df["_qty"] = pd.to_numeric(df["Cant. Completada (Cripto)"], errors="coerce").fillna(0)
    82‚Üí    82‚Üí    df["_qty_usdt"] = pd.to_numeric(df["Cant. Completada (Monto)"], errors="coerce").fillna(0)
    83‚Üí    83‚Üí    df["_leverage"] = pd.to_numeric(df["Apalancamiento"], errors="coerce").fillna(1)
    84‚Üí    84‚Üí    
    85‚Üí    85‚Üí    df = df.sort_values("_ts").reset_index(drop=True)
    86‚Üí    86‚Üí    
    87‚Üí    87‚Üí    LOG.info("Action distribution: %s", df["_action"].value_counts().to_dict())
    88‚Üí    88‚Üí    LOG.info("Unique symbols: %d", df["_symbol"].nunique())
    89‚Üí    89‚Üí    LOG.info("Time range: %s ‚Üí %s", df["_ts"].min(), df["_ts"].max())
    90‚Üí    90‚Üí    
    91‚Üí    91‚Üí    return df

    94‚Üí    94‚Üídef match_trades(orders: pd.DataFrame) -> pd.DataFrame:
    95‚Üí    95‚Üí    """Match open/close orders into trades (FIFO matching)."""
    96‚Üí    96‚Üí    LOG.info("Matching open/close orders into trades...")
    97‚Üí    97‚Üí    
    98‚Üí    98‚Üí    trades = []
    99‚Üí    99‚Üí    opens = defaultdict(lambda: defaultdict(list))
   100‚Üí   100‚Üí    
   101‚Üí   101‚Üí    for idx, row in orders.iterrows():
   102‚Üí   102‚Üí        action = row.get("_action")
   103‚Üí   103‚Üí        if not action or pd.isna(action):
   104‚Üí   104‚Üí            continue
   105‚Üí   105‚Üí        
   106‚Üí   106‚Üí        symbol = row["_symbol"]
   107‚Üí   107‚Üí        ts = row["_ts"]
   108‚Üí   108‚Üí        price = row["_price"]
   109‚Üí   109‚Üí        pnl = row["_pnl"]
   110‚Üí   110‚Üí        fee = row["_fee"]
   111‚Üí   111‚Üí        qty = row["_qty"]
   112‚Üí   112‚Üí        leverage = row["_leverage"]
   113‚Üí   113‚Üí        
   114‚Üí   114‚Üí        if "open" in action:
   115‚Üí   115‚Üí            direction = "long" if "long" in action else "short"
   116‚Üí   116‚Üí            opens[symbol][direction].append({
   117‚Üí   117‚Üí                "ts": ts, "price": price, "fee": fee, "qty": qty, 
   118‚Üí   118‚Üí                "leverage": leverage, "idx": idx
   119‚Üí   119‚Üí            })
   120‚Üí   120‚Üí        
   121‚Üí   121‚Üí        elif "close" in action:
   122‚Üí   122‚Üí            direction = "long" if "long" in action else "short"
   123‚Üí   123‚Üí            open_list = opens[symbol][direction]
   124‚Üí   124‚Üí            
   125‚Üí   125‚Üí            if not open_list:
   126‚Üí   126‚Üí                trades.append({
   127‚Üí   127‚Üí                    "symbol": symbol, "direction": direction,
   128‚Üí   128‚Üí                    "entry_time": ts - timedelta(minutes=30),
   129‚Üí   129‚Üí                    "entry_price": price, "exit_time": ts, "exit_price": price,
   130‚Üí   130‚Üí                    "pnl": pnl, "fee": fee, "n_entries": 1, "leverage": leverage,
   131‚Üí   131‚Üí                })
   132‚Üí   132‚Üí                continue
   133‚Üí   133‚Üí            
   134‚Üí   134‚Üí            # FIFO matching
   135‚Üí   135‚Üí            matched = []
   136‚Üí   136‚Üí            remaining = qty
   137‚Üí   137‚Üí            while open_list and remaining > 0:
   138‚Üí   138‚Üí                o = open_list.pop(0)
   139‚Üí   139‚Üí                matched.append(o)
   140‚Üí   140‚Üí                remaining -= o["qty"]
   141‚Üí   141‚Üí            
   142‚Üí   142‚Üí            if remaining < 0 and matched:
   143‚Üí   143‚Üí                partial = matched.pop()
   144‚Üí   144‚Üí                partial["qty"] = -remaining
   145‚Üí   145‚Üí                open_list.insert(0, partial)
   146‚Üí   146‚Üí            
   147‚Üí   147‚Üí            if matched:
   148‚Üí   148‚Üí                first = matched[0]
   149‚Üí   149‚Üí                total_fee = sum(o["fee"] for o in matched) + fee
   150‚Üí   150‚Üí                avg_entry = np.average([o["price"] for o in matched],
   151‚Üí   151‚Üí                                       weights=[o["qty"] for o in matched])
   152‚Üí   152‚Üí                trades.append({
   153‚Üí   153‚Üí                    "symbol": symbol, "direction": direction,
   154‚Üí   154‚Üí                    "entry_time": first["ts"], "entry_price": avg_entry,
   155‚Üí   155‚Üí                    "exit_time": ts, "exit_price": price,
   156‚Üí   156‚Üí                    "pnl": pnl, "fee": total_fee,
   157‚Üí   157‚Üí                    "n_entries": len(matched), "leverage": first.get("leverage", leverage),
   158‚Üí   158‚Üí                })
   159‚Üí   159‚Üí    
   160‚Üí   160‚Üí    # Remaining open positions
   161‚Üí   161‚Üí    for sym, dirs in opens.items():
   162‚Üí   162‚Üí        for direction, open_list in dirs.items():
   163‚Üí   163‚Üí            for o in open_list:
   164‚Üí   164‚Üí                trades.append({
   165‚Üí   165‚Üí                    "symbol": sym, "direction": direction,
   166‚Üí   166‚Üí                    "entry_time": o["ts"], "entry_price": o["price"],
   167‚Üí   167‚Üí                    "exit_time": pd.NaT, "exit_price": 0,
   168‚Üí   168‚Üí                    "pnl": 0, "fee": o["fee"],
   169‚Üí   169‚Üí                    "n_entries": 1, "leverage": o.get("leverage", 1),
   170‚Üí   170‚Üí                })
   171‚Üí   171‚Üí    
   172‚Üí   172‚Üí    result = pd.DataFrame(trades)
   173‚Üí   173‚Üí    closed = result[result["pnl"] != 0].copy()
   174‚Üí   174‚Üí    LOG.info("Matched %d total, %d closed (with PnL)", len(result), len(closed))
   175‚Üí   175‚Üí    if len(closed) > 0:
   176‚Üí   176‚Üí        LOG.info("Closed: WR=%.1f%% PnL=%.2f", (closed["pnl"] > 0).mean()*100, closed["pnl"].sum())
   177‚Üí   177‚Üí    return result
   178‚Üí   178‚Üí

   184‚Üí   184‚Üídef download_ohlcv(symbol: str, timeframe: str, start_ts: int, end_ts: int) -> pd.DataFrame:
   185‚Üí   185‚Üí    """Download with disk caching."""
   186‚Üí   186‚Üí    import ccxt
   187‚Üí   187‚Üí    
   188‚Üí   188‚Üí    cache_file = CACHE_DIR / f"{symbol}_{timeframe}.parquet"
   189‚Üí   189‚Üí    
   190‚Üí   190‚Üí    if cache_file.exists():
   191‚Üí   191‚Üí        cached = pd.read_parquet(cache_file)
   192‚Üí   192‚Üí        if len(cached) > 0 and cached["timestamp"].min() <= start_ts + 86400000 and cached["timestamp"].max() >= end_ts - 86400000:
   193‚Üí   193‚Üí            return cached[(cached["timestamp"] >= start_ts) & (cached["timestamp"] <= end_ts)].copy()
   194‚Üí   194‚Üí    
   195‚Üí   195‚Üí    for exchange_id in ["binance", "mexc"]:
   196‚Üí   196‚Üí        try:
   197‚Üí   197‚Üí            exchange = getattr(ccxt, exchange_id)({"enableRateLimit": True})
   198‚Üí   198‚Üí            ccxt_sym = f"{symbol}/USDT"
   199‚Üí   199‚Üí            exchange.load_markets()
   200‚Üí   200‚Üí            if ccxt_sym not in exchange.markets:
   201‚Üí   201‚Üí                continue
   202‚Üí   202‚Üí            
   203‚Üí   203‚Üí            all_ohlcv = []
   204‚Üí   204‚Üí            since = start_ts
   205‚Üí   205‚Üí            while True:
   206‚Üí   206‚Üí                ohlcv = exchange.fetch_ohlcv(ccxt_sym, timeframe, since=since, limit=1000)
   207‚Üí   207‚Üí                if not ohlcv:
   208‚Üí   208‚Üí                    break
   209‚Üí   209‚Üí                all_ohlcv.extend(ohlcv)
   210‚Üí   210‚Üí                last_ts = ohlcv[-1][0]
   211‚Üí   211‚Üí                if last_ts >= end_ts or len(ohlcv) < 1000:
   212‚Üí   212‚Üí                    break
   213‚Üí   213‚Üí                since = last_ts + 1
   214‚Üí   214‚Üí                time.sleep(exchange.rateLimit / 1000)
   215‚Üí   215‚Üí            
   216‚Üí   216‚Üí            if all_ohlcv:
   217‚Üí   217‚Üí                df = pd.DataFrame(all_ohlcv, columns=["timestamp", "open", "high", "low", "close", "volume"])
   218‚Üí   218‚Üí                df = df.drop_duplicates(subset=["timestamp"]).sort_values("timestamp").reset_index(drop=True)
   219‚Üí   219‚Üí                df = df[(df["timestamp"] >= start_ts) & (df["timestamp"] <= end_ts)]
   220‚Üí   220‚Üí                df.to_parquet(cache_file, index=False)
   221‚Üí   221‚Üí                LOG.info("  %s: %d bars from %s (cached)", symbol, len(df), exchange_id)
   222‚Üí   222‚Üí                return df
   223‚Üí   223‚Üí        except Exception as e:
   224‚Üí   224‚Üí            LOG.warning("  %s on %s: %s", symbol, exchange_id, str(e)[:80])
   225‚Üí   225‚Üí            time.sleep(2)
   226‚Üí   226‚Üí    
   227‚Üí   227‚Üí    LOG.warning("  %s: no data available", symbol)
   228‚Üí   228‚Üí    return pd.DataFrame()

   235‚Üí   235‚Üídef compute_pattern_features(closes, highs, lows, opens, volumes, i, lookback=50):
   236‚Üí   236‚Üí    """Compute ~30 pattern detector features at bar i (only past data)."""
   237‚Üí   237‚Üí    n = len(closes)
   238‚Üí   238‚Üí    if i < lookback or i >= n:
   239‚Üí   239‚Üí        return None
   240‚Üí   240‚Üí    
   241‚Üí   241‚Üí    c = closes[:i+1]
   242‚Üí   242‚Üí    h = highs[:i+1]
   243‚Üí   243‚Üí    l = lows[:i+1]
   244‚Üí   244‚Üí    o = opens[:i+1]
   245‚Üí   245‚Üí    v = volumes[:i+1]
   246‚Üí   246‚Üí    f = {}
   247‚Üí   247‚Üí    
   248‚Üí   248‚Üí    # ‚îÄ‚îÄ BREAKOUT ‚îÄ‚îÄ
   249‚Üí   249‚Üí    high_20 = np.max(h[-20:])
   250‚Üí   250‚Üí    low_20 = np.min(l[-20:])
   251‚Üí   251‚Üí    range_20 = high_20 - low_20
   252‚Üí   252‚Üí    if range_20 > 0:
   253‚Üí   253‚Üí        f["close_position_20"] = (c[-1] - low_20) / range_20
   254‚Üí   254‚Üí        f["is_breakout_up"] = float(c[-1] >= high_20)
   255‚Üí   255‚Üí        f["is_breakout_down"] = float(c[-1] <= low_20)
   256‚Üí   256‚Üí        f["breakout_strength"] = max(0, abs(c[-1] - (high_20 if c[-1] >= high_20 else low_20)) / range_20)
   257‚Üí   257‚Üí    else:
   258‚Üí   258‚Üí        f["close_position_20"] = 0.5; f["is_breakout_up"] = 0; f["is_breakout_down"] = 0; f["breakout_strength"] = 0
   259‚Üí   259‚Üí    
   260‚Üí   260‚Üí    # ‚îÄ‚îÄ EMA BOUNCE ‚îÄ‚îÄ
   261‚Üí   261‚Üí    ema9 = _ema(c, 9); ema21 = _ema(c, 21); ema50 = _ema(c, 50) if len(c) >= 50 else _ema(c, min(len(c), 21))
   262‚Üí   262‚Üí    atr = _atr(h, l, c, 14)
   263‚Üí   263‚Üí    
   264‚Üí   264‚Üí    if atr > 0:
   265‚Üí   265‚Üí        f["dist_ema9_atr"] = (c[-1] - ema9[-1]) / atr
   266‚Üí   266‚Üí        f["dist_ema21_atr"] = (c[-1] - ema21[-1]) / atr
   267‚Üí   267‚Üí        f["dist_ema50_atr"] = (c[-1] - ema50[-1]) / atr
   268‚Üí   268‚Üí    else:
   269‚Üí   269‚Üí        f["dist_ema9_atr"] = 0; f["dist_ema21_atr"] = 0; f["dist_ema50_atr"] = 0
   270‚Üí   270‚Üí    
   271‚Üí   271‚Üí    f["ema21_bounce"] = 0.0; f["ema50_bounce"] = 0.0
   272‚Üí   272‚Üí    if atr > 0 and len(c) >= 6:
   273‚Üí   273‚Üí        for j in range(1, min(4, len(c))):
   274‚Üí   274‚Üí            if l[-j] <= ema21[-j] + 0.3*atr and h[-j] >= ema21[-j] - 0.3*atr:
   275‚Üí   275‚Üí                f["ema21_bounce"] = 1.0; break
   276‚Üí   276‚Üí        for j in range(1, min(6, len(c))):
   277‚Üí   277‚Üí            if l[-j] <= ema50[-j] + 0.3*atr and h[-j] >= ema50[-j] - 0.3*atr:
   278‚Üí   278‚Üí                f["ema50_bounce"] = 1.0; break
   279‚Üí   279‚Üí    
   280‚Üí   280‚Üí    f["ema_alignment"] = float(np.sign(ema9[-1] - ema21[-1]))
   281‚Üí   281‚Üí    f["ema_trend_strength"] = (ema9[-1] - ema50[-1]) / max(abs(ema50[-1]), 1e-10) * 100
   282‚Üí   282‚Üí    
   283‚Üí   283‚Üí    # ‚îÄ‚îÄ SQUEEZE ‚îÄ‚îÄ
   284‚Üí   284‚Üí    if len(c) >= 20:
   285‚Üí   285‚Üí        sma20 = np.mean(c[-20:]); std20 = np.std(c[-20:])
   286‚Üí   286‚Üí        bb_width = 4*std20 / max(sma20, 1e-10)
   287‚Üí   287‚Üí        f["bb_width"] = bb_width
   288‚Üí   288‚Üí        bb_list = []
   289‚Üí   289‚Üí        for j in range(1, min(30, len(c)-20)):
   290‚Üí   290‚Üí            seg = c[-(20+j):-(j)]
   291‚Üí   291‚Üí            if len(seg) >= 20:
   292‚Üí   292‚Üí                s, sd = np.mean(seg), np.std(seg)
   293‚Üí   293‚Üí                if s > 0: bb_list.append(4*sd/s)
   294‚Üí   294‚Üí        f["squeeze_score"] = 1.0 - np.mean([w <= bb_width for w in bb_list]) if bb_list else 0.5
   295‚Üí   295‚Üí    else:
   296‚Üí   296‚Üí        f["bb_width"] = 0; f["squeeze_score"] = 0.5
   297‚Üí   297‚Üí    
   298‚Üí   298‚Üí    # ‚îÄ‚îÄ VOLUME ‚îÄ‚îÄ
   299‚Üí   299‚Üí    if len(v) >= 20:
   300‚Üí   300‚Üí        vol_ma = np.mean(v[-20:])
   301‚Üí   301‚Üí        f["vol_ratio"] = v[-1] / max(vol_ma, 1e-10)
   302‚Üí   302‚Üí        f["vol_ratio_3"] = np.mean(v[-3:]) / max(vol_ma, 1e-10)
   303‚Üí   303‚Üí    else:
   304‚Üí   304‚Üí        f["vol_ratio"] = 1.0; f["vol_ratio_3"] = 1.0
   305‚Üí   305‚Üí    
   306‚Üí   306‚Üí    # ‚îÄ‚îÄ CANDLE PATTERNS ‚îÄ‚îÄ
   307‚Üí   307‚Üí    body = c[-1] - o[-1]
   308‚Üí   308‚Üí    bar_range = max(h[-1] - l[-1], 1e-10)
   309‚Üí   309‚Üí    f["body_ratio"] = abs(body) / bar_range
   310‚Üí   310‚Üí    f["lower_wick_ratio"] = (min(o[-1], c[-1]) - l[-1]) / bar_range
   311‚Üí   311‚Üí    f["upper_wick_ratio"] = (h[-1] - max(o[-1], c[-1])) / bar_range
   312‚Üí   312‚Üí    f["is_doji"] = float(f["body_ratio"] < 0.1)
   313‚Üí   313‚Üí    f["is_hammer"] = float(f["lower_wick_ratio"] > 0.4 and f["upper_wick_ratio"] < 0.15)
   314‚Üí   314‚Üí    f["is_shooting_star"] = float(f["upper_wick_ratio"] > 0.4 and f["lower_wick_ratio"] < 0.15)
   315‚Üí   315‚Üí    f["is_bull_pin"] = float(f["lower_wick_ratio"] > 0.6 and f["body_ratio"] < 0.3)
   316‚Üí   316‚Üí    f["is_bear_pin"] = float(f["upper_wick_ratio"] > 0.6 and f["body_ratio"] < 0.3)
   317‚Üí   317‚Üí    
   318‚Üí   318‚Üí    if len(c) >= 2:
   319‚Üí   319‚Üí        prev_body = c[-2] - o[-2]
   320‚Üí   320‚Üí        f["is_bullish_engulf"] = float(body > 0 and prev_body < 0 and c[-1] > o[-2] and o[-1] < c[-2])
   321‚Üí   321‚Üí        f["is_bearish_engulf"] = float(body < 0 and prev_body > 0 and c[-1] < o[-2] and o[-1] > c[-2])
   322‚Üí   322‚Üí    else:
   323‚Üí   323‚Üí        f["is_bullish_engulf"] = 0; f["is_bearish_engulf"] = 0
   324‚Üí   324‚Üí    
   325‚Üí   325‚Üí    # ‚îÄ‚îÄ PULLBACK ‚îÄ‚îÄ
   326‚Üí   326‚Üí    trend_dir = np.sign(ema9[-1] - ema50[-1])
   327‚Üí   327‚Üí    high_6 = np.max(h[-6:]) if len(h) >= 6 else np.max(h)
   328‚Üí   328‚Üí    low_6 = np.min(l[-6:]) if len(l) >= 6 else np.min(l)
   329‚Üí   329‚Üí    if high_6 > low_6:
   330‚Üí   330‚Üí        if trend_dir > 0:
   331‚Üí   331‚Üí            f["pullback_depth"] = (high_6 - c[-1]) / (high_6 - low_6)
   332‚Üí   332‚Üí            f["is_pullback_long"] = float(f["pullback_depth"] > 0.3 and trend_dir > 0)
   333‚Üí   333‚Üí            f["is_pullback_short"] = 0.0
   334‚Üí   334‚Üí        elif trend_dir < 0:
   335‚Üí   335‚Üí            f["pullback_depth"] = (c[-1] - low_6) / (high_6 - low_6)
   336‚Üí   336‚Üí            f["is_pullback_short"] = float(f["pullback_depth"] > 0.3 and trend_dir < 0)
   337‚Üí   337‚Üí            f["is_pullback_long"] = 0.0
   338‚Üí   338‚Üí        else:
   339‚Üí   339‚Üí            f["pullback_depth"] = 0.5; f["is_pullback_long"] = 0; f["is_pullback_short"] = 0
   340‚Üí   340‚Üí    else:
   341‚Üí   341‚Üí        f["pullback_depth"] = 0.5; f["is_pullback_long"] = 0; f["is_pullback_short"] = 0
   342‚Üí   342‚Üí    
   343‚Üí   343‚Üí    # ‚îÄ‚îÄ MOMENTUM ‚îÄ‚îÄ
   344‚Üí   344‚Üí    f["ret_1_pct"] = (c[-1] - c[-2]) / max(c[-2], 1e-10)*100 if len(c) >= 2 else 0
   345‚Üí   345‚Üí    f["ret_3_pct"] = (c[-1] - c[-4]) / max(c[-4], 1e-10)*100 if len(c) >= 4 else 0
   346‚Üí   346‚Üí    f["ret_6_pct"] = (c[-1] - c[-7]) / max(c[-7], 1e-10)*100 if len(c) >= 7 else 0
   347‚Üí   347‚Üí    
   348‚Üí   348‚Üí    if len(c) > 10:
   349‚Üí   349‚Üí        rets = np.diff(c[-50:]) / np.maximum(c[-50:-1], 1e-10) if len(c) > 50 else np.diff(c) / np.maximum(c[:-1], 1e-10)
   350‚Üí   350‚Üí        f["momentum_z"] = (f["ret_1_pct"]/100) / max(np.std(rets), 1e-10) if len(rets) > 5 else 0
   351‚Üí   351‚Üí    else:
   352‚Üí   352‚Üí        f["momentum_z"] = 0
   353‚Üí   353‚Üí    
   354‚Üí   354‚Üí    consec = 0
   355‚Üí   355‚Üí    for j in range(1, min(10, len(c)-1)):
   356‚Üí   356‚Üí        if (c[-j] - c[-j-1]) * np.sign(f["ret_1_pct"] + 1e-20) > 0: consec += 1
   357‚Üí   357‚Üí        else: break
   358‚Üí   358‚Üí    f["consecutive_bars"] = consec * np.sign(f["ret_1_pct"])
   359‚Üí   359‚Üí    
   360‚Üí   360‚Üí    # ‚îÄ‚îÄ LEVEL TEST ‚îÄ‚îÄ
   361‚Üí   361‚Üí    high_50 = np.max(h[-50:]) if len(h) >= 50 else np.max(h)
   362‚Üí   362‚Üí    low_50 = np.min(l[-50:]) if len(l) >= 50 else np.min(l)
   363‚Üí   363‚Üí    f["near_high_50"] = float(abs(c[-1] - high_50) / max(high_50, 1e-10) < 0.01)
   364‚Üí   364‚Üí    f["near_low_50"] = float(abs(c[-1] - low_50) / max(low_50, 1e-10) < 0.01)
   365‚Üí   365‚Üí    
   366‚Üí   366‚Üí    # ‚îÄ‚îÄ RANGE EXPANSION ‚îÄ‚îÄ
   367‚Üí   367‚Üí    if len(h) >= 3:
   368‚Üí   368‚Üí        range_3 = np.mean([h[-j] - l[-j] for j in range(1, min(4, len(h)+1))])
   369‚Üí   369‚Üí        range_20r = np.mean([h[-j] - l[-j] for j in range(1, min(21, len(h)+1))])
   370‚Üí   370‚Üí        f["range_expansion"] = range_3 / max(range_20r, 1e-10)
   371‚Üí   371‚Üí    else:
   372‚Üí   372‚Üí        f["range_expansion"] = 1.0
   373‚Üí   373‚Üí    
   374‚Üí   374‚Üí    # ‚îÄ‚îÄ V-REVERSAL ‚îÄ‚îÄ
   375‚Üí   375‚Üí    if len(c) >= 7:
   376‚Üí   376‚Üí        r1 = (c[-4] - c[-7]) / max(c[-7], 1e-10)
   377‚Üí   377‚Üí        r2 = (c[-1] - c[-4]) / max(c[-4], 1e-10)
   378‚Üí   378‚Üí        f["v_reversal"] = float(np.sign(r1) != np.sign(r2) and abs(r2) > abs(r1)*0.5)
   379‚Üí   379‚Üí    else:
   380‚Üí   380‚Üí        f["v_reversal"] = 0.0
   381‚Üí   381‚Üí    
   382‚Üí   382‚Üí    # ‚îÄ‚îÄ META ‚îÄ‚îÄ
   383‚Üí   383‚Üí    f["atr_pct"] = atr / max(c[-1], 1e-10) * 100
   384‚Üí   384‚Üí    
   385‚Üí   385‚Üí    return f

   387‚Üí   387‚Üídef _ema(data, period):
   388‚Üí   388‚Üí    if len(data) < period: return np.full(len(data), data[-1] if len(data) > 0 else 0)
   389‚Üí   389‚Üí    alpha = 2 / (period + 1)
   390‚Üí   390‚Üí    result = np.zeros(len(data)); result[0] = data[0]
   391‚Üí   391‚Üí    for i in range(1, len(data)): result[i] = alpha * data[i] + (1 - alpha) * result[i-1]
   392‚Üí   392‚Üí    return result

   394‚Üí   394‚Üídef _atr(highs, lows, closes, period=14):
   395‚Üí   395‚Üí    if len(closes) < 2: return 0.0
   396‚Üí   396‚Üí    n = min(period, len(closes) - 1)
   397‚Üí   397‚Üí    trs = []
   398‚Üí   398‚Üí    for i in range(max(1, len(closes) - n), len(closes)):
   399‚Üí   399‚Üí        tr = max(highs[i] - lows[i], abs(highs[i] - closes[i-1]), abs(lows[i] - closes[i-1]))
   400‚Üí   400‚Üí        trs.append(tr)
   401‚Üí   401‚Üí    return np.mean(trs) if trs else 0.0

   408‚Üí   408‚Üídef classify_pattern(feat: dict) -> str:
   409‚Üí   409‚Üí    """Classify entry into a visual pattern type."""
   410‚Üí   410‚Üí    if (feat.get("is_breakout_up") or feat.get("is_breakout_down")) and feat.get("vol_ratio", 1) > 1.3:
   411‚Üí   411‚Üí        return "BREAKOUT_UP" if feat.get("is_breakout_up") else "BREAKOUT_DOWN"
   412‚Üí   412‚Üí    if feat.get("breakout_strength", 0) > 0.1:
   413‚Üí   413‚Üí        return "BREAKOUT_UP" if feat.get("close_position_20", 0.5) > 0.9 else "BREAKOUT_DOWN"
   414‚Üí   414‚Üí    if feat.get("ema21_bounce") or feat.get("ema50_bounce"):
   415‚Üí   415‚Üí        return "EMA_BOUNCE_LONG" if feat.get("ema_alignment", 0) > 0 else "EMA_BOUNCE_SHORT"
   416‚Üí   416‚Üí    if feat.get("is_pullback_long"): return "PULLBACK_LONG"
   417‚Üí   417‚Üí    if feat.get("is_pullback_short"): return "PULLBACK_SHORT"
   418‚Üí   418‚Üí    if feat.get("squeeze_score", 0) > 0.7 and feat.get("range_expansion", 1) > 1.5:
   419‚Üí   419‚Üí        return "SQUEEZE_BREAK"
   420‚Üí   420‚Üí    if feat.get("v_reversal"):
   421‚Üí   421‚Üí        return "V_REVERSAL_UP" if feat.get("ret_1_pct", 0) > 0 else "V_REVERSAL_DOWN"
   422‚Üí   422‚Üí    if feat.get("is_bullish_engulf"): return "ENGULFING_BULL"
   423‚Üí   423‚Üí    if feat.get("is_bearish_engulf"): return "ENGULFING_BEAR"
   424‚Üí   424‚Üí    if feat.get("is_hammer"): return "HAMMER"
   425‚Üí   425‚Üí    if feat.get("is_shooting_star"): return "SHOOTING_STAR"
   426‚Üí   426‚Üí    if feat.get("is_bull_pin"): return "PIN_BAR_BULL"
   427‚Üí   427‚Üí    if feat.get("is_bear_pin"): return "PIN_BAR_BEAR"
   428‚Üí   428‚Üí    if abs(feat.get("momentum_z", 0)) > 2.0:
   429‚Üí   429‚Üí        return "IMPULSE_UP" if feat.get("momentum_z") > 0 else "IMPULSE_DOWN"
   430‚Üí   430‚Üí    if feat.get("near_low_50"): return "SUPPORT_TEST"
   431‚Üí   431‚Üí    if feat.get("near_high_50"): return "RESISTANCE_TEST"
   432‚Üí   432‚Üí    return "NO_PATTERN"
