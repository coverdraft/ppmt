   101→def _sax_symbol_to_str(sym) -> str:
   102→    """Convert a SAX symbol (str or tuple) to a pure string.
   103→    
   104→    Single: 'a' → 'a'
   105→    Dual:   ('a', 'x') → 'a'  (price symbol only, for path IDs)
   106→    """
   107→    if isinstance(sym, tuple):
   108→        return str(sym[0])  # price symbol
   109→    return str(sym)
   110→
   111→
   112→def _sax_symbol_to_json(sym) -> list[str] | str:
   113→    """Convert a SAX symbol to JSON-safe format.
   114→    
   115→    Single: 'a' → 'a'
   116→    Dual:   ('a', 'x') → ['a', 'x']  (pure JSON string array)
   117→    """
   118→    if isinstance(sym, tuple):
   119→        return [str(s) for s in sym]
   120→    return str(sym)

   122→
   123→def _build_active_path_ids(backward_path: list) -> list[str]:
   124→    """Build cumulative path IDs from a backward path.
   125→    
   126→    backward_path: ['a', 'b', 'c'] or [('a','x'), ('b','y'), ('c','z')]
   127→    Result: ["root", "a", "a-b", "a-b-c"]
   128→    """
   129→    ids = ["root"]
   130→    cumulative = []
   131→    for sym in backward_path:
   132→        key = _sax_symbol_to_str(sym)
   133→        cumulative.append(key)
   134→        ids.append("-".join(cumulative))
   135→    return ids

   344→    Runs in a background thread. Results are streamed to the client

   355→
   356→    if days < 1 or days > 30:
   357→        return {"status": "error", "message": f"Days must be 1-30, got {days}"}
   358→
   359→    # Prevent duplicate backtests for the same symbol:tf
   360→    bt_key = f"{symbol}:{timeframe}"
   361→    if bt_key in _BACKTEST_RUNNING:
   362→        return {"status": "error", "message": f"Backtest already running for {symbol} {timeframe}"}
   363→
   364→    # Check if WS client is registered
   365→    ws = _BACKTEST_WS.get(bt_key)
   366→    if not ws:
   367→        logger.warning(f"[BACKTEST] No WS client registered for {bt_key}")
   368→        return {"status": "error", "message": f"No WebSocket connection for {symbol} {timeframe}. Connect to the terminal first."}
   369→
   370→    logger.info(f"[BACKTEST] Starting: {symbol} {timeframe} {days}d (WS client found)")
   371→
   372→    asyncio.create_task(_run_backtest_async(symbol, timeframe, days))
   373→
   374→    return {"status": "started", "symbol": symbol, "timeframe": timeframe, "days": days}

   376→
   377→async def _run_backtest_async(symbol: str, timeframe: str, days: int):
   378→    """Run the backtest in a thread and stream results via WebSocket in real-time."""
   379→    bt_key = f"{symbol}:{timeframe}"
   380→    _BACKTEST_RUNNING.add(bt_key)
   381→    msg_queue: queue.Queue = queue.Queue()
   382→
   383→    bt_thread = threading.Thread(
   384→        target=_backtest_sync,
   385→        args=(symbol, timeframe, days, msg_queue),
   386→        daemon=True,
   387→    )
   388→    bt_thread.start()
   389→    logger.info(f"[BACKTEST] Thread started for {bt_key}")
   390→
   391→    try:
   392→        # Drain the queue and send messages in real-time via WS
   393→        got_complete = False
   394→        while True:
   395→            # Thread finished AND queue empty → done
   396→            if not bt_thread.is_alive() and msg_queue.empty():
   397→                break
   398→
   399→            try:
   400→                msg = msg_queue.get(timeout=0.3)
   401→            except queue.Empty:
   402→                await asyncio.sleep(0.05)  # Yield to event loop
   403→                continue
   404→
   405→            # Send the message via WS
   406→            ws = _BACKTEST_WS.get(bt_key)
   407→            if ws:
   408→                try:
   409→                    await ws.send_json(msg)
   410→                except Exception as e:
   411→                    logger.error(f"[BACKTEST] WS send failed: {e}")
   412→                    _BACKTEST_WS.pop(bt_key, None)
   413→                    break
   414→            else:
   415→                logger.warning(f"[BACKTEST] WS client gone for {bt_key}, stopping stream")
   416→                break
   417→
   418→            # If this was the completion message, we're done
   419→            if msg.get("type") == "backtest_complete":
   420→                got_complete = True
   421→                break
   422→
   423→        # If thread died without sending backtest_complete, send error
   424→        if not got_complete:
   425→            logger.error(f"[BACKTEST] Thread ended without sending complete for {bt_key}")
   426→            ws = _BACKTEST_WS.get(bt_key)
   427→            if ws:
   428→                try:
   429→                    await ws.send_json({"type": "backtest_complete", "data": {
   430→                        "error": "Backtest thread crashed (no completion message)",
   431→                        "trades": 0, "wins": 0, "losses": 0,
   432→                        "wr": 0, "pnl_pct": 0, "profit_factor": 0, "max_drawdown": 0,
   433→                    }})
   434→                except Exception:
   435→                    _BACKTEST_WS.pop(bt_key, None)
   436→
   437→    except Exception as e:
   438→        logger.error(f"[BACKTEST] Async streamer failed: {e}", exc_info=True)
   439→        ws = _BACKTEST_WS.get(bt_key)
   440→        if ws:
   441→            try:
   442→                await ws.send_json({"type": "backtest_complete", "data": {
   443→                    "error": str(e), "trades": 0, "wins": 0, "losses": 0,
   444→                    "wr": 0, "pnl_pct": 0, "profit_factor": 0, "max_drawdown": 0,
   445→                }})
   446→            except Exception:
   447→                _BACKTEST_WS.pop(bt_key, None)
   448→    finally:
   449→        _BACKTEST_RUNNING.discard(bt_key)
   450→        logger.info(f"[BACKTEST] Finished: {bt_key}")

   453→def _backtest_sync(symbol: str, timeframe: str, days: int, msg_queue: queue.Queue) -> None:
   454→    """Synchronous backtest logic — runs in a thread.
   455→
   456→    This function replicates EXACTLY the same logic as full_replay_v21.py:
   457→    - Loads IS + OOS data (DB first, Binance REST fallback)
   458→    - Builds tries from IS data with Config F alpha=3
   459→    - Runs OOS replay with Config F parameters
   460→    - Streams results via msg_queue
   461→
   462→    CRITICAL: ALL SQLite connections must be created INSIDE this function
   463→    because SQLite objects can only be used in the thread that created them.
   464→    """
   465→    import copy
   466→    import requests as _requests
   467→    from ppmt.data.classifier import AssetClassifier
   468→    from ppmt.data.storage import PPMTStorage as _PPMTStorage, UNIVERSAL_POOL_KEY, class_pool_key
   469→    from ppmt.core.trie import PPMTTrie, RegimePartitionedTrie
   470→    from ppmt.core.regime import RegimeDetector
   471→    from ppmt.core.profiles import SPREAD_ESTIMATES
   472→    from ppmt.core.thresholds import TIMEFRAME_HARD_MOVE_FLOOR
   473→    from ppmt.engine.weights import AdaptiveWeights
   474→    from ppmt.core.sax import LEVEL_DUAL_ALPHA_CONFIG, LEVEL_DUAL_ALPHA_TF_OVERRIDES
   475→
   476→    def _send(msg_type: str, data: dict):
   477→        msg_queue.put({"type": msg_type, "data": data})
   478→
   479→    # Config F parameters (EXACTLY matching full_replay_v21.py Config F)
   480→    EV_THRESHOLD = 0.40
   481→    SL_MULT = 2.0
   482→    ALPHA_N3_N4 = 3  # Config F uses alpha=3
   483→    CONFIG_F_WEIGHTS = {"n1": 0.10, "n2": 0.00, "n3": 0.90, "n4": 0.00, "n5": 0.00}
   484→    HARD_MOVE_FLOOR = 0.10  # Config F: 0.10% for 5m
   485→    CAPITAL_USDT = 1000.0
   486→    RISK_PCT = 0.01
   487→    IS_DAYS = 60  # In-sample period for trie building
   488→    REGIME_WINDOW_SIZE = 10
   489→
   490→    try:
   491→        logger.info(f"[BACKTEST] _backtest_sync started: {symbol} {timeframe} {days}d")
   492→
   493→        # 1. Classify asset
   494→        classifier = AssetClassifier()
   495→        info = classifier.classify(symbol)
   496→        asset_class = info.asset_class
   497→        weight_profile = info.weight_profile
   498→        logger.info(f"[BACKTEST] Classified: {symbol} → {asset_class}/{weight_profile}")
   499→
   500→        # 2. Load data — try DB first, fall back to Binance download
   501→        storage = _PPMTStorage()  # NEW instance in this thread (SQLite safe)
   502→        total_days = IS_DAYS + days
   503→        df = None
   504→
   505→        # Try loading from DB
   506→        try:
   507→            db_df = storage.load_ohlcv(symbol, timeframe)
   508→            if db_df is not None and len(db_df) > 1000:
   509→                logger.info(f"[BACKTEST] DB has {len(db_df)} candles for {symbol} {timeframe}")
   510→                df = db_df
   511→        except Exception as e:
   512→            logger.warning(f"[BACKTEST] DB load failed: {e}")
   513→
   514→        # Fall back to Binance REST API (direct, like full_replay_v21.py)
   515→        if df is None:
   516→            logger.info(f"[BACKTEST] Fetching {total_days}d of {timeframe} data from Binance REST API...")
   517→            api_symbol = symbol.replace("/", "")
   518→            tf_ms = {"1m": 60000, "5m": 300000, "15m": 900000, "1h": 3600000}
   519→            candle_ms = tf_ms.get(timeframe, 300000)
   520→            target_candles = total_days * (86400000 // candle_ms)
   521→
