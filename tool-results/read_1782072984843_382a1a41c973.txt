    89→# ============================================================
    90→# v0.34.0: TF-aware recalibration interval
    91→# ============================================================
    92→# 15m es la referencia (2000 velas ≈ 21 días).
    93→# TFs más altos → menos recalibraciones (más espaciadas).
    94→# TFs más bajos → usar el base (no tiene sentido recalibrar cada 200 velas).
    95→# Techo de 50k para TFs muy altos (4h, 1d) — mantiene Living Trie vivo
    96→# sin forzar recalibraciones absurdas (526 años para 1d).
    97→_RECALIBRATION_CEILING = 50_000
    98→_RECALIBRATION_BASE = 2_000
    99→_RECALIBRATION_REF_TF_MIN = 15
   100→
   101→_TF_TO_MINUTES = {
   102→    "1m": 1, "3m": 3, "5m": 5, "10m": 10, "15m": 15, "30m": 30,
   103→    "1h": 60, "2h": 120, "4h": 240, "6h": 360, "8h": 480, "12h": 720,
   104→    "1d": 1440, "3d": 4320, "1w": 10080, "1M": 43200,
   105→}

   107→
   108→def _tf_to_minutes(tf: str) -> int:
   109→    """Convierte '15m' -> 15, '1h' -> 60, '1d' -> 1440. Default 15."""
   110→    return _TF_TO_MINUTES.get(tf, 15)

   112→
   113→def get_recalibration_interval(tf_minutes: int) -> int:
   114→    """Calcula el intervalo de recalibración óptimo según el TF.
   115→
   116→    Tabla:
   117→      1m  → 2,000  (33h)
   118→      5m  → 2,000  (7d)
   119→      15m → 2,000  (21d)  ← referencia
   120→      1h  → 8,000  (333d)
   121→      4h  → 32,000 (1333d)
   122→      1d  → 50,000 (526d, capped)
   123→    """
   124→    if tf_minutes <= 0:
   125→        return _RECALIBRATION_BASE
   126→    factor = max(1.0, tf_minutes / _RECALIBRATION_REF_TF_MIN)
   127→    interval = int(_RECALIBRATION_BASE * factor)
   128→    return min(interval, _RECALIBRATION_CEILING)

   339→# v0.40.33: Direct HTTP polling using requests+asyncio.to_thread.
   340→# v0.40.32 used aiohttp but it timed out silently on some networks
   341→# (asyncio.TimeoutError has empty str repr → "Failed to connect: ").
   342→# requests uses the same network stack as the working /api/market/price
   343→# endpoint (ccxt sync + requests), so behavior is now consistent.
   344→class _DirectPollExchange:
   345→    """Lightweight exchange wrapper using requests + asyncio.to_thread.
   346→
   347→    Implements only the methods the engine actually uses:
   348→    - fetch_ticker(symbol) -> {last, close, ...}
   349→    - fetch_ohlcv(symbol, timeframe, limit) -> [[ts, o, h, l, c, v], ...]
   350→    - close() -> no-op (requests creates a new session per call, no cleanup needed)
   351→
   352→    Supports: mexc, binance, bybit (spot). All HTTP calls go through the
   353→    `requests` library wrapped in asyncio.to_thread, so we get the same
   354→    network behavior as sync ccxt calls (which work on the user's Mac).
   355→    """
   356→
   357→    _BASE_URLS = {
   358→        "mexc": "https://api.mexc.com",
   359→        "binance": "https://api.binance.com",
   360→        "bybit": "https://api.bybit.com",
   361→    }
   362→
   363→    _TIMEFRAME_MAP = {
   364→        # CCXT-style -> exchange-native
   365→        "mexc": {
   366→            "1m": "1m", "5m": "5m", "15m": "15m", "30m": "30m",
   367→            "1h": "1H", "4h": "4H", "1d": "1D",
   368→        },
   369→        "binance": {
   370→            "1m": "1m", "5m": "5m", "15m": "15m", "30m": "30m",
   371→            "1h": "1h", "4h": "4h", "1d": "1d",
   372→        },
   373→        "bybit": {
   374→            "1m": "1", "5m": "5", "15m": "15", "30m": "30",
   375→            "1h": "60", "4h": "240", "1d": "D",
   376→        },
   377→    }
   378→
   379→    def __init__(self, exchange_name: str, api_key: str = None, api_secret: str = None):
   380→        import requests  # noqa: F401 — sanity check that requests is installed
   381→        self.exchange_name = exchange_name.lower()
   382→        self.api_key = api_key
   383→        self.api_secret = api_secret
   384→        self.base_url = self._BASE_URLS.get(self.exchange_name)
   385→        if not self.base_url:
   386→            raise ValueError(f"_DirectPollExchange: unsupported exchange '{exchange_name}'")
   387→        # ccxt-compat fields
   388→        self.markets = {}
   389→        self.markets_by_id = {}
   390→        # Shared requests session for connection pooling
   391→        self._session = requests.Session()
   392→        self._session.headers.update({"User-Agent": "ppmt/0.40.33"})
   393→
   394→    def _sync_fetch_ticker(self, symbol: str) -> dict:
   395→        """Sync impl — runs in thread via to_thread."""
   396→        symbol_id = symbol.replace("/", "").upper()
   397→        if self.exchange_name in ("mexc", "binance"):
   398→            url = f"{self.base_url}/api/v3/ticker/price"
   399→            params = {"symbol": symbol_id}
   400→        elif self.exchange_name == "bybit":
   401→            url = f"{self.base_url}/v5/market/tickers"
   402→            params = {"category": "spot", "symbol": symbol_id}
   403→        else:
   404→            raise ValueError(f"fetch_ticker not implemented for {self.exchange_name}")
   405→        resp = self._session.get(url, params=params, timeout=15)
   406→        if resp.status_code != 200:
   407→            raise RuntimeError(f"{self.exchange_name} fetch_ticker HTTP {resp.status_code}: {resp.text[:300]}")
   408→        data = resp.json()
   409→        if self.exchange_name in ("mexc", "binance"):
   410→            price = float(data.get("price", 0))
   411→            return {"symbol": symbol, "last": price, "close": price, "info": data}
   412→        elif self.exchange_name == "bybit":
   413→            tickers = data.get("result", {}).get("list", [])
   414→            if not tickers:
   415→                raise RuntimeError(f"bybit fetch_ticker: no data for {symbol}")
   416→            price = float(tickers[0].get("lastPrice", 0))
   417→            return {"symbol": symbol, "last": price, "close": price, "info": data}
   418→
   419→    async def fetch_ticker(self, symbol: str) -> dict:
   420→        """Async wrapper around sync impl via asyncio.to_thread."""
   421→        import asyncio
   422→        return await asyncio.to_thread(self._sync_fetch_ticker, symbol)
   423→
   424→    def _sync_fetch_ohlcv(self, symbol: str, timeframe: str, limit: int = 100) -> list:
   425→        """Sync impl — runs in thread via to_thread."""
   426→        symbol_id = symbol.replace("/", "").upper()
   427→        tf_native = self._TIMEFRAME_MAP.get(self.exchange_name, {}).get(timeframe, timeframe)
   428→        if self.exchange_name in ("mexc", "binance"):
   429→            url = f"{self.base_url}/api/v3/klines"
   430→            params = {"symbol": symbol_id, "interval": tf_native, "limit": limit}
   431→        elif self.exchange_name == "bybit":
   432→            url = f"{self.base_url}/v5/market/kline"
   433→            params = {"category": "spot", "symbol": symbol_id, "interval": tf_native, "limit": limit}
   434→        else:
   435→            raise ValueError(f"fetch_ohlcv not implemented for {self.exchange_name}")
   436→        resp = self._session.get(url, params=params, timeout=15)
   437→        if resp.status_code != 200:
   438→            raise RuntimeError(f"{self.exchange_name} fetch_ohlcv HTTP {resp.status_code}: {resp.text[:300]}")
   439→        data = resp.json()
   440→        if self.exchange_name in ("mexc", "binance"):
   441→            # [[ts, o, h, l, c, v, ...], ...]
   442→            return [[int(c[0]), float(c[1]), float(c[2]), float(c[3]), float(c[4]), float(c[5])] for c in data]
   443→        elif self.exchange_name == "bybit":
   444→            klines = data.get("result", {}).get("list", [])
   445→            # Bybit returns newest-first, reverse to oldest-first
   446→            klines = list(reversed(klines))
   447→            return [[int(c[0]), float(c[1]), float(c[2]), float(c[3]), float(c[4]), float(c[5])] for c in klines]
   448→
   449→    async def fetch_ohlcv(self, symbol: str, timeframe: str, limit: int = 100) -> list:
   450→        """Async wrapper around sync impl via asyncio.to_thread."""
   451→        import asyncio
   452→        return await asyncio.to_thread(self._sync_fetch_ohlcv, symbol, timeframe, limit)
   453→
   454→    async def close(self):
   455→        # No-op — requests.Session closes itself on GC. Kept for API compat
   456→        # with the engine's `await poll_exchange.close()` calls.
   457→        try:
   458→            self._session.close()
   459→        except Exception:
   460→            pass

   466→class RealtimeTrader:
   467→    """
   468→    Real-Time Trading Engine with incremental SAX encoding.
   469→
   470→    Unlike PaperTrader which batch-processes all data, RealtimeTrader
   471→    processes candles one at a time through the incremental SAX pipeline,
   472→    maintaining a streaming pattern buffer and generating signals only
   473→    when new SAX symbols are produced (every window_size candles).
   474→
   475→    This mirrors production behavior where future data is unavailable.
   476→

   621→                        grid = {

   631→                            best_alpha=cal_alpha,
   632→                            best_window=cal_window,
   633→                            metric=cal_best_result.trading_metric if cal_best_result else 0.0,
   634→                            grid=grid,

   639→                                      f"(PnL={cal_best_result.total_pnl_pct:+.1f}%, "
   640→                                      f"WR={cal_best_result.win_rate:.1%})")
   641→                    else:
   642→                        console.print(f"[green]Calibration confirms: "
   643→                                      f"α={cal_alpha}/W={cal_window}[/green]")
   644→                except Exception as e:
   645→                    console.print(f"[yellow]Calibration failed: {e} — "
   646→                                  f"using defaults[/yellow]")
   647→
   648→            # Save calibrated profile
   649→            if token_profile is not None and token_profile.calibration_date:
   650→                try:

   652→                except Exception:

   667→        return token_profile, cfg

   669→    def _compute_atr_pct(self, prices: np.ndarray, highs: np.ndarray,
   670→                         lows: np.ndarray, period: int = 14) -> float:
   671→        """Compute current ATR as percentage of price.
   672→
   673→        v0.20.0: Fixed to use Wilder's smoothing (same as paper_trader.py).
   674→        Previously used simple moving average which gave inconsistent ATR values
   675→        compared to the backtest engine.
   676→        """
   677→        if len(prices) < period + 1:
   678→            return 2.0  # Default
   679→
   680→        close = prices
   681→        high = highs
   682→        low = lows
   683→
   684→        prev_close = np.roll(close, 1)
   685→        prev_close[0] = close[0]
   686→        tr = np.maximum(
   687→            high - low,
   688→            np.maximum(
   689→                np.abs(high - prev_close),
   690→                np.abs(low - prev_close)
   691→            )
   692→        )
   693→
   694→        # v0.20.0: Wilder's smoothing (exponential moving average)
   695→        # This matches paper_trader.py compute_atr_pct()
   696→        if len(tr) >= period:
   697→            atr_val = np.mean(tr[:period])
   698→            for i in range(period, len(tr)):
   699→                atr_val = (atr_val * (period - 1) + tr[i]) / period
   700→        else:
   701→            atr_val = np.mean(tr)
   702→
   703→        last_price = prices[-1] if prices[-1] > 0 else 1.0
   704→        return (atr_val / last_price * 100) if last_price > 0 else 2.0

   706→    def run_replay(self) -> RealtimeResult:
   707→        """
   708→        Run replay mode: step through historical data as if it were live.
   709→
   710→        Each candle is processed through the incremental SAX encoder.
   711→        When a new SAX symbol is produced (every window_size candles),
   712→        the pattern buffer is updated and a prediction is generated.
   713→
   714→        v0.11.0: Now uses TokenProfile, auto-calibration, FuzzyMatcher,
   715→        4-level matching, Living Trie, and prediction-aware SL/TP.
   716→
   717→        Returns:
   718→            RealtimeResult with trading statistics
   719→        """
   720→        if not isinstance(self.config, ReplayConfig):
   721→            raise ValueError("run_replay() requires ReplayConfig")
   722→
   723→        cfg = self.config
   724→        start_time = time.time()
   725→        storage = PPMTStorage()
   726→
   727→        # Load data
   728→        df = storage.load_ohlcv(cfg.symbol, cfg.timeframe)
   729→        if df.empty:
   730→            console.print(f"[red]No data for {cfg.symbol}. Run 'ppmt ingest' first.[/red]")
   731→            return RealtimeResult(mode="replay", symbol=cfg.symbol, timeframe=cfg.timeframe)
   732→
   733→        # Classify asset
   734→        classifier = AssetClassifier()
   735→        info = classifier.classify(cfg.symbol)
   736→
   737→        # v0.11.0: TokenProfile + auto-calibration + persistence
   738→        token_profile, cfg = self._setup_token_profile(cfg, info, storage, df)
   739→
   740→        # Load or build tries (v0.11.0: all 4 levels)
   741→        # v0.40.4 FIX-1D: pass asset_class so N1/N2 load from cross-asset pools.
   742→        from ppmt.engine.ppmt import PPMT
   743→        all_tries = storage.load_all_tries(cfg.symbol, asset_class=info.asset_class)
   744→        trie_n1 = all_tries["n1"]
   745→        trie_n2 = all_tries["n2"]
   746→        trie_n3 = all_tries["n3"]
   747→        trie_n4 = all_tries["n4"]
   748→
   749→        has_multi_level = (
   750→            getattr(cfg, 'use_multi_level', False)
   751→            and trie_n1 is not None
   752→            and trie_n2 is not None
   753→            and trie_n4 is not None
   754→        )
   755→
   756→        if trie_n3 is None:
   757→            console.print(f"[yellow]No Trie for {cfg.symbol}. Building from data...[/yellow]")
   758→            engine = PPMT(
   759→                symbol=cfg.symbol,
   760→                asset_class=info.asset_class,
   761→                sax_alphabet_size=cfg.sax_alphabet_size,
   762→                sax_window_size=cfg.sax_window_size,
   763→                sax_strategy=cfg.sax_strategy,
   764→                fuzzy_threshold=token_profile.fuzzy_threshold if token_profile else 0.80,
   765→                weight_profile=info.weight_profile,
   766→            )
   767→            # FIX-1D: contribute to universal N1 + class N2 pools
   768→            engine.attach_storage(storage)
   769→            engine.build(df, pattern_length=cfg.pattern_length)
   770→            # Reload N1/N2 from cross-asset pools (engine.trie_n1/2 are empty
   771→            # in storage mode)
   772→            # v0.41.0 FIX-1B: DO NOT fall back to engine.trie_n1/n2 — those
   773→            # are empty in storage mode and would mask a missing shared pool.
   774→            # Empty pools correctly get 0 confidence, allowing weight
   775→            # redistribution to N2/N3 via AdaptiveWeights.
   776→            all_tries = storage.load_all_tries(cfg.symbol, asset_class=info.asset_class)
   777→            trie_n1 = all_tries.get("n1")
   778→            trie_n2 = all_tries.get("n2")
   779→            if trie_n1 is None:
   780→                console.print("[yellow]WARNING: N1 universal pool is empty after build. Only N2/N3/N4 available.[/yellow]")
   781→                trie_n1 = PPMTTrie(name="universal_empty")
   782→            if trie_n2 is None:
   783→                console.print(f"[yellow]WARNING: N2 class pool for {info.asset_class} is empty after build.[/yellow]")
   784→                trie_n2 = PPMTTrie(name=f"class_empty:{info.asset_class}")
   785→            trie_n3 = engine.trie_n3
   786→            trie_n4 = engine.trie_n4
   787→            has_multi_level = True
   788→        else:
   789→            console.print(f"[green]Loaded N3 Trie for {cfg.symbol} ({trie_n3.pattern_count} patterns)[/green]")
   790→            # v0.41.0 FIX-1B: Replace None N1/N2 with empty tries instead of
   791→            # silently dropping to single-level mode.
   792→            if trie_n1 is None:
   793→                console.print("[yellow]WARNING: N1 universal pool not found. Using empty trie.[/yellow]")
   794→                trie_n1 = PPMTTrie(name="universal_empty")
   795→            if trie_n2 is None:
   796→                console.print(f"[yellow]WARNING: N2 class pool for {info.asset_class} not found. Using empty trie.[/yellow]")
   797→                trie_n2 = PPMTTrie(name=f"class_empty:{info.asset_class}")

   828→        if len(warmup_df) >= cfg.sax_window_size * 2:
   829→            try:

   851→            trie,
   852→            prediction_depth=cfg.pattern_length,
   853→            regime_trie=trie_n4 if (has_multi_level and trie_n4 is not None) else None,
   854→        )

   860→            ppmt_engine = PPMT(
   861→                symbol=cfg.symbol,
   862→                asset_class=info.asset_class,
   863→                sax_alphabet_size=cfg.sax_alphabet_size,
   864→                sax_window_size=cfg.sax_window_size,
   865→                sax_strategy=cfg.sax_strategy,
   866→                fuzzy_threshold=fuzzy_threshold,
   867→                weight_profile=info.weight_profile,
   868→            )
   869→            ppmt_engine.set_tries(trie_n1, trie_n2, trie_n3, trie_n4)

   871→            console.print(f"  [bold cyan]4-level matching enabled[/bold cyan]: weights={ppmt_engine.weights}")
   872→
   873→        # Regime detector
   874→        regime_detector = None
   875→        current_regime = "ranging"
   876→        regime_info = None
   877→        if cfg.regime_aware:

   904→        consecutive_breaks = 0
   905→        last_losing_trade_idx = -999
   906→
   907→        # v0.46.0: Pattern Divergence Monitor & BTC Context Filter
   908→        divergence_monitor = PatternDivergenceMonitor(divergence_threshold=0.667)
   909→        btc_filter = BTCContextFilter()
