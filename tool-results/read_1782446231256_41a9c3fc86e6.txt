    50тЖТ    50тЖТ    LOG.info("Loading MEXC orders from %s", xlsx_path.name)

    54тЖТ    54тЖТ    # Parse timestamps (MEXC gives UTC+2, convert to UTC)

    60тЖТ    60тЖТ

    64тЖТ    64тЖТ    #   sell short = OPEN SHORT  (PNL always 0)
    65тЖТ    65тЖТ    #   buy short = CLOSE SHORT  (PNL = trade result)
    66тЖТ    66тЖТ    df["_action"] = df["Direcci├│n"].str.strip().str.lower().map({
    67тЖТ    67тЖТ        "buy long": "open_long",
    68тЖТ    68тЖТ        "sell long": "close_long",

    90тЖТ    90тЖТdef match_trades(orders: pd.DataFrame) -> pd.DataFrame:
    91тЖТ    91тЖТ    """Match open/close orders into trades (FIFO matching)."""
    92тЖТ    92тЖТ    LOG.info("Matching open/close orders into trades...")
    93тЖТ    93тЖТ    
    94тЖТ    94тЖТ    trades = []
    95тЖТ    95тЖТ    opens = defaultdict(lambda: defaultdict(list))
    96тЖТ    96тЖТ    
    97тЖТ    97тЖТ    for idx, row in orders.iterrows():
    98тЖТ    98тЖТ        action = row.get("_action")
    99тЖТ    99тЖТ        if not action or pd.isna(action):
   100тЖТ   100тЖТ            continue
   101тЖТ   101тЖТ        
   102тЖТ   102тЖТ        symbol = row["_symbol"]
   103тЖТ   103тЖТ        ts = row["_ts"]
   104тЖТ   104тЖТ        price = row["_price"]
   105тЖТ   105тЖТ        pnl = row["_pnl"]
   106тЖТ   106тЖТ        fee = row["_fee"]
   107тЖТ   107тЖТ        qty = row["_qty"]
   108тЖТ   108тЖТ        leverage = row["_leverage"]
   109тЖТ   109тЖТ        
   110тЖТ   110тЖТ        if "open" in action:
   111тЖТ   111тЖТ            direction = "long" if "long" in action else "short"
   112тЖТ   112тЖТ            opens[symbol][direction].append({
   113тЖТ   113тЖТ                "ts": ts, "price": price, "fee": fee, "qty": qty, 
   114тЖТ   114тЖТ                "leverage": leverage, "idx": idx
   115тЖТ   115тЖТ            })
   116тЖТ   116тЖТ        
   117тЖТ   117тЖТ        elif "close" in action:
   118тЖТ   118тЖТ            direction = "long" if "long" in action else "short"
   119тЖТ   119тЖТ            open_list = opens[symbol][direction]
   120тЖТ   120тЖТ            
   121тЖТ   121тЖТ            if not open_list:
   122тЖТ   122тЖТ                trades.append({
   123тЖТ   123тЖТ                    "symbol": symbol, "direction": direction,
   124тЖТ   124тЖТ                    "entry_time": ts - timedelta(minutes=30),
   125тЖТ   125тЖТ                    "entry_price": price, "exit_time": ts, "exit_price": price,
   126тЖТ   126тЖТ                    "pnl": pnl, "fee": fee, "n_entries": 1, "leverage": leverage,
   127тЖТ   127тЖТ                })
   128тЖТ   128тЖТ                continue
   129тЖТ   129тЖТ            
   130тЖТ   130тЖТ            # FIFO matching
   131тЖТ   131тЖТ            matched = []
   132тЖТ   132тЖТ            remaining = qty
   133тЖТ   133тЖТ            while open_list and remaining > 0:
   134тЖТ   134тЖТ                o = open_list.pop(0)
   135тЖТ   135тЖТ                matched.append(o)
   136тЖТ   136тЖТ                remaining -= o["qty"]
   137тЖТ   137тЖТ            
   138тЖТ   138тЖТ            if remaining < 0 and matched:
   139тЖТ   139тЖТ                partial = matched.pop()
   140тЖТ   140тЖТ                partial["qty"] = -remaining
   141тЖТ   141тЖТ                open_list.insert(0, partial)
   142тЖТ   142тЖТ            
   143тЖТ   143тЖТ            if matched:
   144тЖТ   144тЖТ                first = matched[0]
   145тЖТ   145тЖТ                total_fee = sum(o["fee"] for o in matched) + fee
   146тЖТ   146тЖТ                avg_entry = np.average([o["price"] for o in matched],
   147тЖТ   147тЖТ                                       weights=[o["qty"] for o in matched])
   148тЖТ   148тЖТ                trades.append({
   149тЖТ   149тЖТ                    "symbol": symbol, "direction": direction,
   150тЖТ   150тЖТ                    "entry_time": first["ts"], "entry_price": avg_entry,
   151тЖТ   151тЖТ                    "exit_time": ts, "exit_price": price,
   152тЖТ   152тЖТ                    "pnl": pnl, "fee": total_fee,
   153тЖТ   153тЖТ                    "n_entries": len(matched), "leverage": first.get("leverage", leverage),
   154тЖТ   154тЖТ                })
   155тЖТ   155тЖТ    
   156тЖТ   156тЖТ    # Remaining open positions
   157тЖТ   157тЖТ    for sym, dirs in opens.items():
   158тЖТ   158тЖТ        for direction, open_list in dirs.items():
   159тЖТ   159тЖТ            for o in open_list:
   160тЖТ   160тЖТ                trades.append({
   161тЖТ   161тЖТ                    "symbol": sym, "direction": direction,
   162тЖТ   162тЖТ                    "entry_time": o["ts"], "entry_price": o["price"],
   163тЖТ   163тЖТ                    "exit_time": pd.NaT, "exit_price": 0,
   164тЖТ   164тЖТ                    "pnl": 0, "fee": o["fee"],
   165тЖТ   165тЖТ                    "n_entries": 1, "leverage": o.get("leverage", 1),
   166тЖТ   166тЖТ                })
   167тЖТ   167тЖТ    
   168тЖТ   168тЖТ    result = pd.DataFrame(trades)
   169тЖТ   169тЖТ    closed = result[result["pnl"] != 0].copy()
   170тЖТ   170тЖТ    LOG.info("Matched %d total, %d closed (with PnL)", len(result), len(closed))
   171тЖТ   171тЖТ    if len(closed) > 0:
   172тЖТ   172тЖТ        LOG.info("Closed: WR=%.1f%% PnL=%.2f", (closed["pnl"] > 0).mean()*100, closed["pnl"].sum())
   173тЖТ   173тЖТ    return result

   180тЖТ   180тЖТdef download_ohlcv(symbol: str, timeframe: str, start_ts: int, end_ts: int) -> pd.DataFrame:
   181тЖТ   181тЖТ    """Download with disk caching."""
   182тЖТ   182тЖТ    import ccxt
   183тЖТ   183тЖТ    
   184тЖТ   184тЖТ    cache_file = CACHE_DIR / f"{symbol}_{timeframe}.parquet"
   185тЖТ   185тЖТ    
   186тЖТ   186тЖТ    if cache_file.exists():
   187тЖТ   187тЖТ        cached = pd.read_parquet(cache_file)
   188тЖТ   188тЖТ        if len(cached) > 0 and cached["timestamp"].min() <= start_ts + 86400000 and cached["timestamp"].max() >= end_ts - 86400000:
   189тЖТ   189тЖТ            return cached[(cached["timestamp"] >= start_ts) & (cached["timestamp"] <= end_ts)].copy()
   190тЖТ   190тЖТ    
   191тЖТ   191тЖТ    for exchange_id in ["binance", "mexc"]:
   192тЖТ   192тЖТ        try:
   193тЖТ   193тЖТ            exchange = getattr(ccxt, exchange_id)({"enableRateLimit": True})
   194тЖТ   194тЖТ            ccxt_sym = f"{symbol}/USDT"
   195тЖТ   195тЖТ            exchange.load_markets()
   196тЖТ   196тЖТ            if ccxt_sym not in exchange.markets:
   197тЖТ   197тЖТ                continue
   198тЖТ   198тЖТ            
   199тЖТ   199тЖТ            all_ohlcv = []
   200тЖТ   200тЖТ            since = start_ts
   201тЖТ   201тЖТ            while True:
   202тЖТ   202тЖТ                ohlcv = exchange.fetch_ohlcv(ccxt_sym, timeframe, since=since, limit=1000)
   203тЖТ   203тЖТ                if not ohlcv:
   204тЖТ   204тЖТ                    break
   205тЖТ   205тЖТ                all_ohlcv.extend(ohlcv)
   206тЖТ   206тЖТ                last_ts = ohlcv[-1][0]
   207тЖТ   207тЖТ                if last_ts >= end_ts or len(ohlcv) < 1000:
   208тЖТ   208тЖТ                    break
   209тЖТ   209тЖТ                since = last_ts + 1
   210тЖТ   210тЖТ                time.sleep(exchange.rateLimit / 1000)
   211тЖТ   211тЖТ            
   212тЖТ   212тЖТ            if all_ohlcv:
   213тЖТ   213тЖТ                df = pd.DataFrame(all_ohlcv, columns=["timestamp", "open", "high", "low", "close", "volume"])
   214тЖТ   214тЖТ                df = df.drop_duplicates(subset=["timestamp"]).sort_values("timestamp").reset_index(drop=True)
   215тЖТ   215тЖТ                df = df[(df["timestamp"] >= start_ts) & (df["timestamp"] <= end_ts)]
   216тЖТ   216тЖТ                df.to_parquet(cache_file, index=False)
   217тЖТ   217тЖТ                LOG.info("  %s: %d bars from %s (cached)", symbol, len(df), exchange_id)
   218тЖТ   218тЖТ                return df
   219тЖТ   219тЖТ        except Exception as e:
   220тЖТ   220тЖТ            LOG.warning("  %s on %s: %s", symbol, exchange_id, str(e)[:80])
   221тЖТ   221тЖТ            time.sleep(2)

   223тЖТ   223тЖТ    LOG.warning("  %s: no data available", symbol)

   231тЖТ   231тЖТdef compute_pattern_features(closes, highs, lows, opens, volumes, i, lookback=50):
   232тЖТ   232тЖТ    """Compute ~30 pattern detector features at bar i (only past data)."""
   233тЖТ   233тЖТ    n = len(closes)
   234тЖТ   234тЖТ    if i < lookback or i >= n:
   235тЖТ   235тЖТ        return None
   236тЖТ   236тЖТ    
   237тЖТ   237тЖТ    c = closes[:i+1]
   238тЖТ   238тЖТ    h = highs[:i+1]
   239тЖТ   239тЖТ    l = lows[:i+1]
   240тЖТ   240тЖТ    o = opens[:i+1]
   241тЖТ   241тЖТ    v = volumes[:i+1]
   242тЖТ   242тЖТ    f = {}
   243тЖТ   243тЖТ    
   244тЖТ   244тЖТ    # тФАтФА BREAKOUT тФАтФА
   245тЖТ   245тЖТ    high_20 = np.max(h[-20:])
   246тЖТ   246тЖТ    low_20 = np.min(l[-20:])
   247тЖТ   247тЖТ    range_20 = high_20 - low_20
   248тЖТ   248тЖТ    if range_20 > 0:
   249тЖТ   249тЖТ        f["close_position_20"] = (c[-1] - low_20) / range_20
   250тЖТ   250тЖТ        f["is_breakout_up"] = float(c[-1] >= high_20)
   251тЖТ   251тЖТ        f["is_breakout_down"] = float(c[-1] <= low_20)
   252тЖТ   252тЖТ        f["breakout_strength"] = max(0, abs(c[-1] - (high_20 if c[-1] >= high_20 else low_20)) / range_20)
   253тЖТ   253тЖТ    else:
   254тЖТ   254тЖТ        f["close_position_20"] = 0.5; f["is_breakout_up"] = 0; f["is_breakout_down"] = 0; f["breakout_strength"] = 0
   255тЖТ   255тЖТ    
   256тЖТ   256тЖТ    # тФАтФА EMA BOUNCE тФАтФА

   270тЖТ   270тЖТ            if l[-j] <= ema21[-j] + 0.3*atr and h[-j] >= ema21[-j] - 0.3*atr:

   272тЖТ   272тЖТ        for j in range(1, min(6, len(c))):

   291тЖТ   291тЖТ    else:

   305тЖТ   305тЖТ    f["body_ratio"] = abs(body) / bar_range

   346тЖТ   346тЖТ        f["momentum_z"] = (f["ret_1_pct"]/100) / max(np.std(rets), 1e-10) if len(rets) > 5 else 0

   373тЖТ   373тЖТ        r2 = (c[-1] - c[-4]) / max(c[-4], 1e-10)
   374тЖТ   374тЖТ        f["v_reversal"] = float(np.sign(r1) != np.sign(r2) and abs(r2) > abs(r1)*0.5)
   375тЖТ   375тЖТ    else:

   383тЖТ   383тЖТdef _ema(data, period):
   384тЖТ   384тЖТ    if len(data) < period: return np.full(len(data), data[-1] if len(data) > 0 else 0)
   385тЖТ   385тЖТ    alpha = 2 / (period + 1)
   386тЖТ   386тЖТ    result = np.zeros(len(data)); result[0] = data[0]
   387тЖТ   387тЖТ    for i in range(1, len(data)): result[i] = alpha * data[i] + (1 - alpha) * result[i-1]
   388тЖТ   388тЖТ    return result

   390тЖТ   390тЖТdef _atr(highs, lows, closes, period=14):
   391тЖТ   391тЖТ    if len(closes) < 2: return 0.0
   392тЖТ   392тЖТ    n = min(period, len(closes) - 1)
   393тЖТ   393тЖТ    trs = []
   394тЖТ   394тЖТ    for i in range(max(1, len(closes) - n), len(closes)):
   395тЖТ   395тЖТ        tr = max(highs[i] - lows[i], abs(highs[i] - closes[i-1]), abs(lows[i] - closes[i-1]))
   396тЖТ   396тЖТ        trs.append(tr)
   397тЖТ   397тЖТ    return np.mean(trs) if trs else 0.0
