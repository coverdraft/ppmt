  1535→// Helper: convert a stored timestamp (string ms, int ms, or int s) to UNIX
  1536→// seconds. Lightweight-Charts requires seconds for intraday timeframes.
  1537→function _toChartTime(t) {
  1538→  if (t === null || t === undefined || t === '') return null;
  1539→  let n = typeof t === 'string' ? parseInt(t, 10) : Number(t);
  1540→  if (!Number.isFinite(n)) {
  1541→    // Try ISO string format (rare but defensive)
  1542→    const d = new Date(t);
  1543→    return isNaN(d.getTime()) ? null : Math.floor(d.getTime() / 1000);
  1544→  }
  1545→  if (n > 1e12) n = Math.floor(n / 1000);  // ms → s
  1546→  return n;
  1547→}

  1549→// Helper: sort markers by time ascending (Lightweight-Charts requirement)
  1550→// and dedupe by (time, text) to avoid overlapping duplicates when a signal
  1551→// and a trade entry fall on the same candle.
  1552→function _mergeAndSortMarkers(signalMarkers, tradeMarkers) {
  1553→  const all = [...signalMarkers, ...tradeMarkers];
  1554→  all.sort((a, b) => (a.time || 0) - (b.time || 0));

  1841→// The WS pushes a snapshot every 1s with `current_price`. We hook into

  1846→let _lastCandleTime = 0;
  1847→function _updateChartLiveTick(price, symbol, timeframe) {
  1848→  if (!chart || !candleSeries || !price || price <= 0) return;
  1849→  const chartSym = (document.getElementById('chartSymbol') || {}).value;
  1850→  if (symbol && chartSym && symbol !== chartSym) return; // not the displayed symbol

  1853→  const secs = tfSecs[tf] || 300;
  1854→  const now = Math.floor(Date.now() / 1000);
  1855→  const bucket = Math.floor(now / secs) * secs;
  1856→  try {
  1857→    const lastBar = candleSeries.data().slice(-1)[0];
  1858→    if (bucket > _lastCandleTime && _lastCandleTime > 0 && lastBar) {
  1859→      // New TF bucket started — push a fresh candle. The previous candle's
  1860→      // close becomes the new candle's open.

  1862→      candleSeries.update(newCandle);
  1863→      _lastCandleTime = bucket;
  1864→    } else if (lastBar && lastBar.time === bucket) {
  1865→      // Same bucket — update last candle's close (and high/low if exceeded)
  1866→      candleSeries.update({
  1867→        time: lastBar.time,
  1868→        open: lastBar.open,
  1869→        high: Math.max(lastBar.high, price),
  1870→        low: Math.min(lastBar.low, price),
  1871→        close: price,
  1872→      });

  1874→      // No prior candle for this bucket — seed one
  1875→      candleSeries.update({ time: bucket, open: price, high: price, low: price, close: price });
