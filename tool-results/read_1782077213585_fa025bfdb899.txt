   316тЖТ   316тЖТ    # тФАтФАтФА Symbol info (lazy) тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   317тЖТ   317тЖТ
   318тЖТ   318тЖТ    async def _fetch_symbol_info(self, symbol: str) -> dict:
   319тЖТ   319тЖТ        """
   320тЖТ   320тЖТ        Fetch contract detail for a single symbol.
   321тЖТ   321тЖТ
   322тЖТ   322тЖТ        GET /api/v1/contract/detail?symbol=DOGE_USDT
   323тЖТ   323тЖТ
   324тЖТ   324тЖТ        Returns and caches:
   325тЖТ   325тЖТ          - price_precision (int): Decimal places for price (e.g. 5)
   326тЖТ   326тЖТ          - quantity_precision (int): Decimal places for quantity (e.g. 0)
   327тЖТ   327тЖТ          - min_qty (float): Minimum order quantity
   328тЖТ   328тЖТ          - contract_size (float): Multiplier per contract
   329тЖТ   329тЖТ        """
   330тЖТ   330тЖТ        mexc_sym = _symbol_to_mexc(symbol)
   331тЖТ   331тЖТ
   332тЖТ   332тЖТ        if mexc_sym in self._symbol_info:
   333тЖТ   333тЖТ            return self._symbol_info[mexc_sym]
   334тЖТ   334тЖТ
   335тЖТ   335тЖТ        resp = await self._request(
   336тЖТ   336тЖТ            "GET",
   337тЖТ   337тЖТ            "/api/v1/contract/detail",
   338тЖТ   338тЖТ            params={"symbol": mexc_sym},
   339тЖТ   339тЖТ            signed=False,
   340тЖТ   340тЖТ        )
   341тЖТ   341тЖТ
   342тЖТ   342тЖТ        data = resp.get("data", {})
   343тЖТ   343тЖТ        info = {
   344тЖТ   344тЖТ            "symbol": mexc_sym,
   345тЖТ   345тЖТ            "price_precision": int(data.get("pricePrecision", 8)),
   346тЖТ   346тЖТ            "quantity_precision": int(data.get("quantityPrecision", 0)),
   347тЖТ   347тЖТ            "min_qty": float(data.get("minQty", 1)),
   348тЖТ   348тЖТ            "contract_size": float(data.get("contractSize", 1)),
   349тЖТ   349тЖТ        }
   350тЖТ   350тЖТ
   351тЖТ   351тЖТ        self._symbol_info[mexc_sym] = info
   352тЖТ   352тЖТ        logger.info(
   353тЖТ   353тЖТ            f"[MEXC] Symbol info cached: {mexc_sym} "
   354тЖТ   354тЖТ            f"price_prec={info['price_precision']} qty_prec={info['quantity_precision']}"
   355тЖТ   355тЖТ        )
   356тЖТ   356тЖТ        return info
   357тЖТ   357тЖТ
   358тЖТ   358тЖТ    # тФАтФАтФА Leverage (lazy) тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   359тЖТ   359тЖТ
   360тЖТ   360тЖТ    async def _ensure_leverage(self, symbol: str, leverage: int) -> None:
   361тЖТ   361тЖТ        """
   362тЖТ   362тЖТ        Set leverage for a symbol if not already set.
   363тЖТ   363тЖТ
   364тЖТ   364тЖТ        POST /api/v1/leverage
   365тЖТ   365тЖТ          symbol=DOGE_USDT&leverage=20&openType=2
   366тЖТ   366тЖТ
   367тЖТ   367тЖТ        Called once per symbol before the first position open.
   368тЖТ   368тЖТ        """
   369тЖТ   369тЖТ        mexc_sym = _symbol_to_mexc(symbol)
   370тЖТ   370тЖТ        if mexc_sym in self._leverage_set:
   371тЖТ   371тЖТ            return
   372тЖТ   372тЖТ
   373тЖТ   373тЖТ        await self._request(
   374тЖТ   374тЖТ            "POST",
   375тЖТ   375тЖТ            "/api/v1/leverage",
   376тЖТ   376тЖТ            params={
   377тЖТ   377тЖТ                "symbol": mexc_sym,
   378тЖТ   378тЖТ                "leverage": leverage,
   379тЖТ   379тЖТ                "openType": 2,  # Cross margin
   380тЖТ   380тЖТ            },
   381тЖТ   381тЖТ        )
   382тЖТ   382тЖТ
   383тЖТ   383тЖТ        self._leverage_set.add(mexc_sym)
   384тЖТ   384тЖТ        logger.info(f"[MEXC] Leverage set: {mexc_sym} = {leverage}x (cross)")
   385тЖТ   385тЖТ
   386тЖТ   386тЖТ    # тФАтФАтФА Quantity calculation тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   387тЖТ   387тЖТ
   388тЖТ   388тЖТ    @staticmethod
   389тЖТ   389тЖТ    def _calc_quantity(
   390тЖТ   390тЖТ        size_usdt: float,
   391тЖТ   391тЖТ        price: float,
   392тЖТ   392тЖТ        leverage: int,
   393тЖТ   393тЖТ        quantity_precision: int,
   394тЖТ   394тЖТ        contract_size: float = 1.0,
   395тЖТ   395тЖТ    ) -> float:
   396тЖТ   396тЖТ        """
   397тЖТ   397тЖТ        Calculate order quantity from USDT size.
   398тЖТ   398тЖТ
   399тЖТ   399тЖТ        quantity = (size_usdt * leverage) / (price * contract_size)
   400тЖТ   400тЖТ
   401тЖТ   401тЖТ        Then round down to quantity_precision decimal places.
   402тЖТ   402тЖТ        """
   403тЖТ   403тЖТ        raw_qty = (size_usdt * leverage) / (price * contract_size)
   404тЖТ   404тЖТ        # Round down to precision (floor)
   405тЖТ   405тЖТ        factor = 10 ** quantity_precision
   406тЖТ   406тЖТ        rounded = int(raw_qty * factor) / factor
   407тЖТ   407тЖТ        return rounded
   408тЖТ   408тЖТ
   409тЖТ   409тЖТ    # тФАтФАтФА IExecutor implementation тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   410тЖТ   410тЖТ
   411тЖТ   411тЖТ    async def open_position(
   412тЖТ   412тЖТ        self,
   413тЖТ   413тЖТ        symbol: str,
   414тЖТ   414тЖТ        direction: str,
   415тЖТ   415тЖТ        size_usdt: float,
   416тЖТ   416тЖТ        metadata: dict,
   417тЖТ   417тЖТ    ) -> PositionState:
   418тЖТ   418тЖТ        """
   419тЖТ   419тЖТ        Open a real futures position on MEXC.
   420тЖТ   420тЖТ
   421тЖТ   421тЖТ        Flow:
   422тЖТ   422тЖТ          1. Fetch symbol precision (lazy, once per symbol).
   423тЖТ   423тЖТ          2. Set leverage (lazy, once per symbol).
   424тЖТ   424тЖТ          3. Calculate quantity from size_usdt.
   425тЖТ   425тЖТ          4. POST /api/v1/order/place-order (market order).
   426тЖТ   426тЖТ          5. Immediately place SL and TP conditional orders.
   427тЖТ   427тЖТ          6. Return PositionState with exchange_meta containing order IDs.
   428тЖТ   428тЖТ        """
   429тЖТ   429тЖТ        if symbol in self._positions and self._positions[symbol].status in (
   430тЖТ   430тЖТ            "ACTIVE", "BREAK_EVEN_SECURED", "TP_EXTENDED"
   431тЖТ   431тЖТ        ):
   432тЖТ   432тЖТ            raise RuntimeError(f"Already in position: {symbol}")
   433тЖТ   433тЖТ
   434тЖТ   434тЖТ        entry_price = metadata.get("entry_price", 0.0)
   435тЖТ   435тЖТ        expected_move_pct = metadata.get("expected_move_pct", 1.0)
   436тЖТ   436тЖТ        leverage = metadata.get("leverage", self._default_leverage)
   437тЖТ   437тЖТ
   438тЖТ   438тЖТ        # 1. Fetch symbol info
   439тЖТ   439тЖТ        info = await self._fetch_symbol_info(symbol)
   440тЖТ   440тЖТ        mexc_sym = info["symbol"]
   441тЖТ   441тЖТ        qty_prec = info["quantity_precision"]
   442тЖТ   442тЖТ        price_prec = info["price_precision"]
   443тЖТ   443тЖТ        contract_size = info["contract_size"]
   444тЖТ   444тЖТ
   445тЖТ   445тЖТ        # 2. Set leverage
   446тЖТ   446тЖТ        await self._ensure_leverage(symbol, leverage)
   447тЖТ   447тЖТ
   448тЖТ   448тЖТ        # 3. Calculate quantity
   449тЖТ   449тЖТ        quantity = self._calc_quantity(
   450тЖТ   450тЖТ            size_usdt=size_usdt,
   451тЖТ   451тЖТ            price=entry_price,
   452тЖТ   452тЖТ            leverage=leverage,
   453тЖТ   453тЖТ            quantity_precision=qty_prec,
   454тЖТ   454тЖТ            contract_size=contract_size,
   455тЖТ   455тЖТ        )
   456тЖТ   456тЖТ
   457тЖТ   457тЖТ        if quantity <= 0:
   458тЖТ   458тЖТ            raise ValueError(
   459тЖТ   459тЖТ                f"Calculated quantity is 0 for {symbol}: "
   460тЖТ   460тЖТ                f"size={size_usdt}, price={entry_price}, leverage={leverage}"
   461тЖТ   461тЖТ            )
   462тЖТ   462тЖТ
   463тЖТ   463тЖТ        # 4. Place market order
   464тЖТ   464тЖТ        side = _direction_to_side(direction)
   465тЖТ   465тЖТ        order_params = {
   466тЖТ   466тЖТ            "symbol": mexc_sym,
   467тЖТ   467тЖТ            "price": round(entry_price, price_prec),
   468тЖТ   468тЖТ            "vol": quantity,
   469тЖТ   469тЖТ            "side": side,
   470тЖТ   470тЖТ            "type": 5,          # 5 = MARKET order
   471тЖТ   471тЖТ            "openType": 2,      # 2 = CROSS margin
   472тЖТ   472тЖТ            "positionType": 1,  # 1 = FIXED position (perpetual)
   473тЖТ   473тЖТ        }
   474тЖТ   474тЖТ
   475тЖТ   475тЖТ        resp = await self._request(
   476тЖТ   476тЖТ            "POST",
   477тЖТ   477тЖТ            "/api/v1/order/place-order",
   478тЖТ   478тЖТ            params=order_params,
   479тЖТ   479тЖТ        )
   480тЖТ   480тЖТ
   481тЖТ   481тЖТ        order_data = resp.get("data", {})
   482тЖТ   482тЖТ        order_id = order_data.get("orderId") or order_data.get("id")
   483тЖТ   483тЖТ
   484тЖТ   484тЖТ        logger.info(
   485тЖТ   485тЖТ            f"[MEXC] Market order placed: {direction} {mexc_sym} "
   486тЖТ   486тЖТ            f"qty={quantity} @ ~{entry_price} order_id={order_id}"
   487тЖТ   487тЖТ        )
   488тЖТ   488тЖТ
   489тЖТ   489тЖТ        # 5. Calculate SL/TP prices
   490тЖТ   490тЖТ        expected_move = entry_price * (expected_move_pct / 100.0)
   491тЖТ   491тЖТ        if direction == "LONG":
   492тЖТ   492тЖТ            sl_price = metadata.get("sl_price", entry_price - (expected_move * 1.2))
   493тЖТ   493тЖТ            tp_price = metadata.get("tp_price", entry_price + (expected_move * 2.5))
   494тЖТ   494тЖТ            cat_sl = entry_price - (expected_move * 3.0)
   495тЖТ   495тЖТ        else:
   496тЖТ   496тЖТ            sl_price = metadata.get("sl_price", entry_price + (expected_move * 1.2))
   497тЖТ   497тЖТ            tp_price = metadata.get("tp_price", entry_price - (expected_move * 2.5))
   498тЖТ   498тЖТ            cat_sl = entry_price + (expected_move * 3.0)
   499тЖТ   499тЖТ
   500тЖТ   500тЖТ        # Round SL/TP to price precision
   501тЖТ   501тЖТ        sl_price = round(sl_price, price_prec)
   502тЖТ   502тЖТ        tp_price = round(tp_price, price_prec)
   503тЖТ   503тЖТ
   504тЖТ   504тЖТ        # Place SL order (stop market)
   505тЖТ   505тЖТ        sl_order_id = None
   506тЖТ   506тЖТ        try:
   507тЖТ   507тЖТ            sl_resp = await self._request(
   508тЖТ   508тЖТ                "POST",
   509тЖТ   509тЖТ                "/api/v1/order/stop-order",
   510тЖТ   510тЖТ                params={
   511тЖТ   511тЖТ                    "symbol": mexc_sym,
   512тЖТ   512тЖТ                    "price": round(sl_price, price_prec),
   513тЖТ   513тЖТ                    "vol": quantity,
   514тЖТ   514тЖТ                    "side": _direction_to_close_side(direction),
   515тЖТ   515тЖТ                    "type": 1,          # 1 = STOP (loss)
   516тЖТ   516тЖТ                    "triggerPrice": round(sl_price, price_prec),
   517тЖТ   517тЖТ                    "triggerType": 1,   # 1 = deal price
   518тЖТ   518тЖТ                    "openType": 2,
   519тЖТ   519тЖТ                },
   520тЖТ   520тЖТ            )
   521тЖТ   521тЖТ            sl_data = sl_resp.get("data", {})
   522тЖТ   522тЖТ            sl_order_id = sl_data.get("orderId") or sl_data.get("id")
   523тЖТ   523тЖТ            logger.info(f"[MEXC] SL order placed: {sl_price} id={sl_order_id}")
   524тЖТ   524тЖТ        except (MexcApiError, Exception) as e:
   525тЖТ   525тЖТ            logger.warning(f"[MEXC] SL order failed (non-fatal): {e}")
   526тЖТ   526тЖТ
   527тЖТ   527тЖТ        # Place TP order (take profit market)
   528тЖТ   528тЖТ        tp_order_id = None
   529тЖТ   529тЖТ        try:
   530тЖТ   530тЖТ            tp_resp = await self._request(
   531тЖТ   531тЖТ                "POST",
   532тЖТ   532тЖТ                "/api/v1/order/stop-order",
   533тЖТ   533тЖТ                params={
   534тЖТ   534тЖТ                    "symbol": mexc_sym,
   535тЖТ   535тЖТ                    "price": round(tp_price, price_prec),
   536тЖТ   536тЖТ                    "vol": quantity,
   537тЖТ   537тЖТ                    "side": _direction_to_close_side(direction),
   538тЖТ   538тЖТ                    "type": 2,          # 2 = TAKE_PROFIT
   539тЖТ   539тЖТ                    "triggerPrice": round(tp_price, price_prec),
   540тЖТ   540тЖТ                    "triggerType": 1,
   541тЖТ   541тЖТ                    "openType": 2,
   542тЖТ   542тЖТ                },
   543тЖТ   543тЖТ            )
   544тЖТ   544тЖТ            tp_data = tp_resp.get("data", {})
   545тЖТ   545тЖТ            tp_order_id = tp_data.get("orderId") or tp_data.get("id")
   546тЖТ   546тЖТ            logger.info(f"[MEXC] TP order placed: {tp_price} id={tp_order_id}")
   547тЖТ   547тЖТ        except (MexcApiError, Exception) as e:
   548тЖТ   548тЖТ            logger.warning(f"[MEXC] TP order failed (non-fatal): {e}")
   549тЖТ   549тЖТ
   550тЖТ   550тЖТ        # 6. Build PositionState
   551тЖТ   551тЖТ        predicted_path = metadata.get("predicted_path_symbols")
   552тЖТ   552тЖТ        expected_sequence = [[s] for s in predicted_path] if predicted_path else []
   553тЖТ   553тЖТ
   554тЖТ   554тЖТ        position = PositionState(
   555тЖТ   555тЖТ            symbol=symbol,
   556тЖТ   556тЖТ            direction=direction,
   557тЖТ   557тЖТ            status="ACTIVE",
   558тЖТ   558тЖТ            entry_price=entry_price,
   559тЖТ   559тЖТ            entry_time=datetime.now(timezone.utc).isoformat(),
   560тЖТ   560тЖТ            size_usdt=size_usdt,
   561тЖТ   561тЖТ            current_sl=sl_price,
   562тЖТ   562тЖТ            current_tp=tp_price,
   563тЖТ   563тЖТ            catastrophic_sl=round(cat_sl, price_prec),
   564тЖТ   564тЖТ            expected_sequence=expected_sequence,
   565тЖТ   565тЖТ            sequence_index=0,
   566тЖТ   566тЖТ            exchange_meta={
   567тЖТ   567тЖТ                "mexc_symbol": mexc_sym,
   568тЖТ   568тЖТ                "order_id": order_id,
   569тЖТ   569тЖТ                "sl_order_id": sl_order_id,
   570тЖТ   570тЖТ                "tp_order_id": tp_order_id,
   571тЖТ   571тЖТ                "quantity": quantity,
   572тЖТ   572тЖТ                "leverage": leverage,
   573тЖТ   573тЖТ                "price_precision": price_prec,
   574тЖТ   574тЖТ                "quantity_precision": qty_prec,
   575тЖТ   575тЖТ            },
   576тЖТ   576тЖТ        )
   577тЖТ   577тЖТ
   578тЖТ   578тЖТ        self._positions[symbol] = position
   579тЖТ   579тЖТ        return position

   581тЖТ   581тЖТ    async def update_position(
   582тЖТ   582тЖТ        self,
   583тЖТ   583тЖТ        position: PositionState,
   584тЖТ   584тЖТ        new_sl: Optional[float] = None,
   585тЖТ   585тЖТ        new_tp: Optional[float] = None,
   586тЖТ   586тЖТ    ) -> bool:
   587тЖТ   587тЖТ        """
   588тЖТ   588тЖТ        Update SL/TP by cancelling old conditional orders and placing new ones.
   589тЖТ   589тЖТ
   590тЖТ   590тЖТ        Golden rule: If cancellation fails (order already filled or triggered),
   591тЖТ   591тЖТ        we log the error and return False тАФ we do NOT crash.
   592тЖТ   592тЖТ        """
   593тЖТ   593тЖТ        if not position.exchange_meta:
   594тЖТ   594тЖТ            logger.warning("[MEXC] update_position: no exchange_meta, cannot update")
   595тЖТ   595тЖТ            return False
   596тЖТ   596тЖТ
   597тЖТ   597тЖТ        mexc_sym = position.exchange_meta.get("mexc_symbol", "")
   598тЖТ   598тЖТ        quantity = position.exchange_meta.get("quantity", 0)
   599тЖТ   599тЖТ        price_prec = position.exchange_meta.get("price_precision", 8)
   600тЖТ   600тЖТ        direction = position.direction
   601тЖТ   601тЖТ        success = True
   602тЖТ   602тЖТ
   603тЖТ   603тЖТ        # Update SL
   604тЖТ   604тЖТ        if new_sl is not None and new_sl != position.current_sl:
   605тЖТ   605тЖТ            old_sl_id = position.exchange_meta.get("sl_order_id")
   606тЖТ   606тЖТ            if old_sl_id:
   607тЖТ   607тЖТ                try:
   608тЖТ   608тЖТ                    await self._request(
   609тЖТ   609тЖТ                        "DELETE",
   610тЖТ   610тЖТ                        "/api/v1/order",
   611тЖТ   611тЖТ                        params={"symbol": mexc_sym, "orderId": old_sl_id},
   612тЖТ   612тЖТ                    )
   613тЖТ   613тЖТ                    logger.info(f"[MEXC] Old SL order cancelled: {old_sl_id}")
   614тЖТ   614тЖТ                except MexcApiError as e:
   615тЖТ   615тЖТ                    logger.warning(f"[MEXC] SL cancel failed (may be filled): {e}")
   616тЖТ   616тЖТ                    success = False
   617тЖТ   617тЖТ
   618тЖТ   618тЖТ            if success:
   619тЖТ   619тЖТ                try:
   620тЖТ   620тЖТ                    sl_resp = await self._request(
   621тЖТ   621тЖТ                        "POST",
   622тЖТ   622тЖТ                        "/api/v1/order/stop-order",
   623тЖТ   623тЖТ                        params={
   624тЖТ   624тЖТ                            "symbol": mexc_sym,
   625тЖТ   625тЖТ                            "price": round(new_sl, price_prec),
   626тЖТ   626тЖТ                            "vol": quantity,
   627тЖТ   627тЖТ                            "side": _direction_to_close_side(direction),
   628тЖТ   628тЖТ                            "type": 1,
   629тЖТ   629тЖТ                            "triggerPrice": round(new_sl, price_prec),
   630тЖТ   630тЖТ                            "triggerType": 1,
   631тЖТ   631тЖТ                            "openType": 2,
   632тЖТ   632тЖТ                        },
   633тЖТ   633тЖТ                    )
   634тЖТ   634тЖТ                    sl_data = sl_resp.get("data", {})
   635тЖТ   635тЖТ                    new_sl_id = sl_data.get("orderId") or sl_data.get("id")
   636тЖТ   636тЖТ                    position.exchange_meta["sl_order_id"] = new_sl_id
   637тЖТ   637тЖТ                    position.current_sl = round(new_sl, price_prec)
   638тЖТ   638тЖТ                    logger.info(f"[MEXC] New SL placed: {new_sl} id={new_sl_id}")
   639тЖТ   639тЖТ                except MexcApiError as e:

   685тЖТ   685тЖТ    async def close_position(
   686тЖТ   686тЖТ        self,
   687тЖТ   687тЖТ        position: PositionState,
   688тЖТ   688тЖТ        reason: str,
   689тЖТ   689тЖТ    ) -> PositionState:
   690тЖТ   690тЖТ        """
   691тЖТ   691тЖТ        Close a specific position on MEXC.
   692тЖТ   692тЖТ
   693тЖТ   693тЖТ        1. Cancel pending SL/TP orders.
   694тЖТ   694тЖТ        2. POST /api/v1/order/close-position with the symbol.
   695тЖТ   695тЖТ        """
   696тЖТ   696тЖТ        if not position.exchange_meta:
   697тЖТ   697тЖТ            raise RuntimeError("No exchange_meta тАФ cannot close position")
   698тЖТ   698тЖТ
   699тЖТ   699тЖТ        mexc_sym = position.exchange_meta.get("mexc_symbol", "")
   700тЖТ   700тЖТ
   701тЖТ   701тЖТ        # Cancel SL
   702тЖТ   702тЖТ        old_sl_id = position.exchange_meta.get("sl_order_id")
   703тЖТ   703тЖТ        if old_sl_id:
   704тЖТ   704тЖТ            try:
   705тЖТ   705тЖТ                await self._request(
   706тЖТ   706тЖТ                    "DELETE",
   707тЖТ   707тЖТ                    "/api/v1/order",
   708тЖТ   708тЖТ                    params={"symbol": mexc_sym, "orderId": old_sl_id},
   709тЖТ   709тЖТ                )
   710тЖТ   710тЖТ            except MexcApiError as e:
   711тЖТ   711тЖТ                logger.warning(f"[MEXC] SL cancel on close (may be filled): {e}")
   712тЖТ   712тЖТ
   713тЖТ   713тЖТ        # Cancel TP
   714тЖТ   714тЖТ        old_tp_id = position.exchange_meta.get("tp_order_id")
   715тЖТ   715тЖТ        if old_tp_id:
   716тЖТ   716тЖТ            try:
   717тЖТ   717тЖТ                await self._request(
   718тЖТ   718тЖТ                    "DELETE",
   719тЖТ   719тЖТ                    "/api/v1/order",
   720тЖТ   720тЖТ                    params={"symbol": mexc_sym, "orderId": old_tp_id},
   721тЖТ   721тЖТ                )
   722тЖТ   722тЖТ            except MexcApiError as e:
   723тЖТ   723тЖТ                logger.warning(f"[MEXC] TP cancel on close (may be filled): {e}")
   724тЖТ   724тЖТ
   725тЖТ   725тЖТ        # Close position
   726тЖТ   726тЖТ        resp = await self._request(
   727тЖТ   727тЖТ            "POST",
   728тЖТ   728тЖТ            "/api/v1/order/close-position",
   729тЖТ   729тЖТ            params={
   730тЖТ   730тЖТ                "symbol": mexc_sym,
   731тЖТ   731тЖТ                "side": _direction_to_close_side(position.direction),
   732тЖТ   732тЖТ            },
   733тЖТ   733тЖТ        )
   734тЖТ   734тЖТ
   735тЖТ   735тЖТ        close_data = resp.get("data", {})
   736тЖТ   736тЖТ        close_price = float(close_data.get("price", position.entry_price))
   737тЖТ   737тЖТ
   738тЖТ   738тЖТ        # Update position
   739тЖТ   739тЖТ        position.close_price = close_price
   740тЖТ   740тЖТ        position.close_reason = reason
   741тЖТ   741тЖТ        position.status = reason
   742тЖТ   742тЖТ
   743тЖТ   743тЖТ        if position.direction == "LONG":
   744тЖТ   744тЖТ            position.pnl_pct = ((close_price - position.entry_price) / position.entry_price) * 100.0
   745тЖТ   745тЖТ        else:
   746тЖТ   746тЖТ            position.pnl_pct = ((position.entry_price - close_price) / position.entry_price) * 100.0
   747тЖТ   747тЖТ
   748тЖТ   748тЖТ        position.pnl_usdt = position.size_usdt * (position.pnl_pct / 100.0)
   749тЖТ   749тЖТ
   750тЖТ   750тЖТ        logger.info(
   751тЖТ   751тЖТ            f"[MEXC] Position closed: {mexc_sym} {reason} @ {close_price} "
   752тЖТ   752тЖТ            f"PnL={position.pnl_pct:+.2f}%"
   753тЖТ   753тЖТ        )
   754тЖТ   754тЖТ
   755тЖТ   755тЖТ        # Remove from active positions
   756тЖТ   756тЖТ        self._positions.pop(position.symbol, None)
   757тЖТ   757тЖТ        return position
