    73→def _get_parent_manager():
    74→    """Get or create the ParentNodeManager."""
    75→    global _parent_manager
    76→    if _parent_manager is None:
    77→        from ppmt.risk.money_manager import ParentNodeManager, ChildNodeConfig
    78→        state_file = os.path.join(CONFIG_DIR, "parent_node_state.json")
    79→        import yaml
    80→
    81→        # Default capital from terminal state
    82→        capital = terminal_state.portfolio_value or 10_000.0
    83→        _parent_manager = ParentNodeManager(total_capital=capital)
    84→
    85→        # Load saved state
    86→        if os.path.exists(state_file):
    87→            try:
    88→                with open(state_file) as f:
    89→                    saved = yaml.safe_load(f) or {}
    90→                _parent_manager.total_capital = saved.get("total_capital", capital)
    91→                for child_data in saved.get("children", []):
    92→                    cfg = ChildNodeConfig(**child_data)
    93→                    _parent_manager.register_child(cfg)
    94→                if _parent_manager._children:
    95→                    _parent_manager.distribute_capital()
    96→            except Exception as e:
    97→                logger.warning(f"Failed to load parent node state: {e}")
    98→
    99→    return _parent_manager

   102→def _save_parent_manager():
   103→    """Save ParentNodeManager state to disk."""
   104→    import yaml
   105→    pm = _get_parent_manager()
   106→    state_file = os.path.join(CONFIG_DIR, "parent_node_state.json")
   107→    try:
   108→        os.makedirs(CONFIG_DIR, exist_ok=True)
   109→        state = {
   110→            "total_capital": pm.total_capital,
   111→            "children": [
   112→                {
   113→                    "node_id": cfg.node_id,
   114→                    "symbol": cfg.symbol,
   115→                    "timeframe": cfg.timeframe,
   116→                    "capital_allocation_pct": cfg.capital_allocation_pct,
   117→                    "leverage": cfg.leverage,
   118→                    "auto_mode": cfg.auto_mode,
   119→                    "max_position_pct": cfg.max_position_pct,
   120→                    "enabled": cfg.enabled,
   121→                }
   122→                for cfg in pm._children.values()
   123→            ],
   124→        }
   125→        with open(state_file, "w") as f:
   126→            yaml.dump(state, f, default_flow_style=False)
   127→    except Exception as e:
   128→        logger.warning(f"Failed to save parent node state: {e}")

   280→@app.post("/api/nodes/add")
   281→async def add_node(req: AddNodeRequest) -> dict:
   282→    """Add a child node."""
   283→    from ppmt.risk.money_manager import ChildNodeConfig
   284→    pm = _get_parent_manager()
   285→    node_id = f"{req.symbol.split('/')[0].lower()}_{req.timeframe}"
   286→
   287→    if node_id in pm._children:
   288→        return {"ok": False, "error": f"Node '{node_id}' already exists"}
   289→
   290→    cfg = ChildNodeConfig(
   291→        node_id=node_id,
   292→        symbol=req.symbol,
   293→        timeframe=req.timeframe,
   294→        capital_allocation_pct=req.capital_allocation_pct,
   295→        leverage=req.leverage,
   296→        auto_mode=req.auto_mode,
   297→    )
   298→
   299→    try:
   300→        pm.register_child(cfg)
   301→        pm.distribute_capital()
   302→        _save_parent_manager()
   303→        return {"ok": True, "node_id": node_id, "capital": pm.get_child_capital(node_id)}
   304→    except Exception as e:
   305→        return {"ok": False, "error": str(e)}

   420→
   421→@app.post("/api/backtest")
   422→async def run_backtest(req: BacktestRequest) -> dict:
   423→    """Run a quick backtest and return results."""
   424→    try:
   425→        from ppmt.engine.realtime import RealtimeTrader, ReplayConfig
   426→        config = ReplayConfig(
   427→            symbol=req.symbol,
   428→            timeframe=req.timeframe,
   429→            initial_capital=req.capital,
   430→            speed=0,
   431→            verbose=False,
   432→        )
   433→        trader = RealtimeTrader(config=config)
   434→        result = trader.run_replay()
   435→
   436→        trades = []
   437→        for t in result.trades:
   438→            # Parse timestamps for chart markers
   439→            entry_ts = None
   440→            exit_ts = None
   441→            try:
   442→                if t.entry_time:
   443→                    import pandas as pd
   444→                    dt = pd.Timestamp(t.entry_time)
   445→                    entry_ts = int(dt.timestamp())
   446→                if t.exit_time:
   447→                    import pandas as pd
   448→                    dt = pd.Timestamp(t.exit_time)
   449→                    exit_ts = int(dt.timestamp())
   450→            except Exception:
   451→                pass
   452→
   453→            trades.append({
   454→                "direction": t.direction,
   455→                "entry_price": t.entry_price,
   456→                "exit_price": t.exit_price,
   457→                "entry_time": t.entry_time,
   458→                "exit_time": t.exit_time,
   459→                "entry_ts": entry_ts,
   460→                "exit_ts": exit_ts,
   461→                "pnl_pct": t.pnl_pct,
   462→                "exit_reason": t.exit_reason,
   463→                "regime": t.regime,
   464→            })
   465→
   466→        return {
   467→            "ok": True,
   468→            "total_trades": result.total_trades,
   469→            "win_rate": result.win_rate,
   470→            "total_pnl_pct": result.total_pnl_pct,
   471→            "max_drawdown": result.max_drawdown,
   472→            "trades": trades,
   473→            "equity_curve": result.equity_curve[-200:] if result.equity_curve else [],
   474→        }
   475→    except Exception as e:
   476→        return {"ok": False, "error": str(e)}

   484→@app.get("/api/ohlcv")
   485→async def get_ohlcv(
   486→    symbol: str = "BTC/USDT",
   487→    timeframe: str = "1m",
   488→    limit: int = 200,
   489→    exchange: str = "mexc",
   490→) -> dict:
   491→    """Fetch real OHLCV data from exchange via ccxt for the candlestick chart."""
   492→    try:
   493→        import ccxt
   494→        ex = getattr(ccxt, exchange, None)
   495→        if ex is None:
   496→            return {"ok": False, "error": f"Exchange '{exchange}' not found"}
   497→        # v0.40.28: spot-only for binance — avoids fapi.binance.com block.
   498→        _opts = {}
   499→        if exchange.lower() == 'binance':
   500→            _opts = {'enableRateLimit': True,
   501→                     'options': {'defaultType': 'spot', 'fetchMarkets': ['spot']}}
   502→        exc = ex(_opts)
   503→        try:
   504→            ohlcv = exc.fetch_ohlcv(symbol, timeframe, limit=min(limit, 1000))
   505→            candles = []
   506→            for c in ohlcv:
   507→                candles.append({
   508→                    "t": c[0],       # timestamp
   509→                    "o": c[1],       # open
   510→                    "h": c[2],       # high
   511→                    "l": c[3],       # low
   512→                    "c": c[4],       # close
   513→                    "v": c[5],       # volume
   514→                })
   515→            return {"ok": True, "symbol": symbol, "timeframe": timeframe, "candles": candles}
   516→        finally:
   517→            if hasattr(exc, 'close'):
   518→                exc.close()
   519→    except Exception as e:
   520→        return {"ok": False, "error": str(e)}

   523→@app.get("/api/market/price")
   524→async def get_market_price(
   525→    symbol: str = "BTC/USDT",
   526→    exchange: str = "mexc",
   527→) -> dict:
   528→    """Get current market price from exchange."""
   529→    try:
   530→        import ccxt
   531→        ex = getattr(ccxt, exchange, None)
   532→        if ex is None:
   533→            return {"ok": False, "error": f"Exchange '{exchange}' not found"}
   534→        # v0.40.28: spot-only for binance — same fapi fix as /api/ohlcv.
   535→        _opts = {}
   536→        if exchange.lower() == 'binance':
   537→            _opts = {'enableRateLimit': True,
   538→                     'options': {'defaultType': 'spot', 'fetchMarkets': ['spot']}}
   539→        exc = ex(_opts)
   540→        try:
   541→            ticker = exc.fetch_ticker(symbol)
   542→            # v0.40.34: Explicit no-cache headers to prevent CDN/browser
   543→            # proxies from serving stale prices (user saw 700-dollar jumps
   544→            # in BTC price suggesting stale cached responses).
   545→            from fastapi.responses import JSONResponse
   546→            return JSONResponse(
   547→                content={
   548→                    "ok": True,
   549→                    "symbol": symbol,
   550→                    "price": ticker.get("last", 0),
   551→                    "change_24h": ticker.get("percentage", 0),
   552→                    "high_24h": ticker.get("high", 0),
   553→                    "low_24h": ticker.get("low", 0),
   554→                    "volume_24h": ticker.get("quoteVolume", 0),
   555→                    "ts": int(time.time()),
   556→                },
   557→                headers={"Cache-Control": "no-store, no-cache, must-revalidate, max-age=0",
   558→                         "Pragma": "no-cache",
   559→                         "Expires": "0"},
   560→            )
   561→        finally:
   562→            if hasattr(exc, 'close'):
   563→                exc.close()
   564→    except Exception as e:
   565→        return {"ok": False, "error": str(e)}

   691→@app.post("/api/start-trading")
   692→async def start_trading(req: StartTradingRequest) -> dict:
   693→    """Start a paper trading session in the background.
   694→
   695→    v0.31.0: Pre-trade gate — checks validation before allowing trading.
   696→    If no recent validation exists, runs auto-setup first.
   697→
   698→    This endpoint performs the full workflow:
   699→    1. Checks pre-trade validation (backtest + MC)
   700→    2. Ingests historical data (if not already present)
   701→    3. Auto-builds Trie (if not already present)
   702→    4. Starts the RealtimeTrader in live/paper mode as a background asyncio task
   703→
   704→    The dashboard WebSocket will receive real-time updates automatically.
   705→    """
   706→    global _trading_task
   707→
   708→    if _trading_task is not None and not _trading_task.done():
   709→        return {"ok": False, "error": "Trading session already running. Stop it first."}
   710→
   711→    # v0.31.0: Pre-trade validation gate
   712→    try:
   713→        storage = PPMTStorage()
   714→        latest_val = storage.get_latest_validation(req.symbol, req.timeframe)
   715→        storage.close()
   716→
   717→        if latest_val is None or latest_val.get("verdict") != "PASS":
   718→            # Auto-validate if no recent validation
   719→            val_result = await validate_token(ValidateRequest(
   720→                symbol=req.symbol,
   721→                timeframe=req.timeframe,
   722→                exchange=req.exchange,
   723→                capital=req.capital,
   724→            ))
   725→            v = val_result.get("verdict")
   726→            if v != "PASS":
   727→                if v == "INSUFFICIENT_DATA":
   728→                    return {
   729→                        "ok": False,
   730→                        "error": f"Cannot trade {req.symbol} {req.timeframe}: backtest produced 0 trades. "
   731→                                 f"Ingest more historical data or try a different timeframe.",
   732→                        "validation": val_result,
   733→                        "checks": val_result.get("checks", {}),
   734→                    }
   735→                else:
   736→                    return {
   737→                        "ok": False,
   738→                        "error": f"Pre-trade validation FAILED for {req.symbol} {req.timeframe}. "
   739→                                 f"Token did not pass safety checks (WR/PF/RoR).",
   740→                        "validation": val_result,
   741→                        "checks": val_result.get("checks", {}),
   742→                    }
   743→    except Exception as e:
   744→        logger.warning(f"Pre-trade gate check failed: {e} — proceeding anyway")
   745→
   746→    try:
   747→        # v0.34.3: Reset terminal state so the new session doesn't show stale
   748→        # pattern_buffer / signals / equity_curve from a previous session.
   749→        # Without this, the dashboard shows 30 'n' (SAX neutral) symbols from
   750→        # the previous run forever, even after Stop+Start.
   751→        terminal_state.reset()
   752→        terminal_state.update_sync(
   753→            symbol=req.symbol,
   754→            timeframe=req.timeframe,
   755→            exchange=req.exchange,
   756→            is_running=True,
   757→            websocket_status="connecting",
   758→            mode="paper",
   759→            started_at=time.time(),
   760→            capital=req.capital,
   761→            leverage=req.leverage,
   762→            auto_mode=req.auto_mode,
   763→        )
   764→
   765→        # Step 1: Auto-ingest data if needed
   766→        storage = PPMTStorage()
   767→        df = storage.load_ohlcv(req.symbol, req.timeframe)
   768→        candles_count = len(df) if df is not None and not df.empty else 0
   769→
   770→        if candles_count < 500:
   771→            logger.info(f"Auto-ingesting {req.days_ingest} days of {req.symbol} {req.timeframe} data...")
   772→            try:
   773→                collector = DataCollector(exchange=req.exchange, storage=storage)
   774→                df = collector.fetch_and_save(req.symbol, req.timeframe, days=req.days_ingest)
   775→                candles_count = len(df) if df is not None and not df.empty else 0
   776→                logger.info(f"Ingested {candles_count} candles")
   777→                # v0.34.3: collector.close() no longer closes the shared storage.
   778→                if hasattr(collector, 'close'):
   779→                    collector.close()
   780→            except Exception as e:
   781→                logger.warning(f"Auto-ingest failed: {e} — continuing with existing data")
   782→
   783→        # v0.34.3: Defensive — re-open storage if it was closed by a child.
   784→        if storage.conn is None:
   785→            try:
   786→                storage._reconnect()
   787→            except Exception:
   788→                storage = PPMTStorage()
   789→
   790→        # Step 2: Auto-build Trie if needed
   791→        # v0.40.4 FIX-1D: pass asset_class so N1/N2 load from shared cross-asset pools.
   792→        _info = AssetClassifier().classify(req.symbol)
   793→        all_tries = storage.load_all_tries(req.symbol, asset_class=_info.asset_class)
   794→        if all_tries.get("n3") is None:
   795→            logger.info(f"Auto-building Trie for {req.symbol} {req.timeframe}...")
   796→            try:
   797→                from ppmt.engine.ppmt import PPMT as PPMTBuilder
   798→                df = storage.load_ohlcv(req.symbol, req.timeframe)
   799→                if df is not None and not df.empty:
   800→                    builder = PPMTBuilder(
   801→                        symbol=req.symbol,
   802→                        asset_class=_info.asset_class,
   803→                        sax_strategy="ohlcv",
   804→                    )
   805→                    # v0.40.4 FIX-1D: attach storage so build() contributes each
   806→                    # observation to the universal N1 pool and the class-shared
   807→                    # N2 pool. Without this, N1/N2 stay empty in production and
   808→                    # 40% of the weighted confidence is wasted (see audit CAPA 1).
   809→                    builder.attach_storage(storage)
   810→                    # Build all 4 levels in one call (persists N1→universal,
   811→                    # N2→class pool, N3→per-symbol, N4→per-symbol+regime)
   812→                    count = builder.build(df, pattern_length=5)
   813→                    logger.info(f"Built {count} patterns; N1 universal + N2 class pools updated")
   814→                    # v0.41.0 FIX-1B: verify shared pools were actually saved
   815→                    pool_status = builder.ensure_shared_pools(storage)
   816→                    if not pool_status["n1_universal"]["exists"]:
   817→                        logger.warning(f"N1 universal pool MISSING after build of {req.symbol}")
   818→                    if not pool_status["n2_class"]["exists"]:
   819→                        logger.warning(f"N2 class pool MISSING after build of {req.symbol}")
   820→                    # Reload via load_all_tries so N1/N2 are the cross-asset
   821→                    # pools (builder.trie_n1/trie_n2 are empty in storage mode).
   822→                    all_tries = storage.load_all_tries(req.symbol, asset_class=_info.asset_class)
   823→                else:
   824→                    storage.close()
   825→                    return {"ok": False, "error": f"No data available for {req.symbol} {req.timeframe}. Ingest failed."}
   826→            except Exception as e:
   827→                logger.warning(f"Auto-build failed: {e}")
   828→                storage.close()
   829→                return {"ok": False, "error": f"Auto-build failed: {str(e)}"}
   830→
   831→        storage.close()
   832→
   833→        # Step 3: Start trading as background task
   834→        _trading_stop_event.clear()
   835→
   836→        async def _run_trading():
   837→            """Background trading task."""
   838→            from ppmt.engine.realtime import RealtimeTrader, LiveConfig
   839→
   840→            config = LiveConfig(
   841→                symbol=req.symbol,
   842→                timeframe=req.timeframe,
   843→                initial_capital=req.capital,
   844→                exchange=req.exchange,
   845→                dry_run=True,  # Always paper trading from dashboard
   846→                testnet=False,  # v0.29.0: Use mainnet data feed (testnet=404 on Binance WS)
   847→                leverage=req.leverage,
   848→                auto_mode=req.auto_mode,
   849→                max_open_positions=req.max_positions,
   850→                max_portfolio_exposure_pct=req.max_exposure,
   851→                kill_switch_pct=req.kill_switch_pct,
   852→                daily_loss_limit_pct=req.daily_loss_pct,
   853→                use_kelly_sizing=True,
   854→                kelly_fraction=0.25,
   855→                use_token_profile=True,
   856→                auto_calibrate=True,
   857→                regime_aware=True,
   858→                use_multi_level=True,
   859→                living_trie=True,
   860→            )
   861→
   862→            trader = RealtimeTrader(config=config)
   863→            try:
   864→                result = await trader.run_live()
   865→                logger.info(f"Trading session ended: {result.total_trades} trades, P&L: {result.total_pnl_pct:.2f}%")
   866→            except Exception as e:
   867→                logger.error(f"Trading session error: {e}")
   868→            finally:
   869→                terminal_state.update_sync(is_running=False, websocket_status="stopped")
   870→
   871→        _trading_task = asyncio.create_task(_run_trading())
   872→
   873→        return {
   874→            "ok": True,
   875→            "message": f"Paper trading started for {req.symbol} {req.timeframe} on {req.exchange}",
   876→            "symbol": req.symbol,
   877→            "timeframe": req.timeframe,
   878→            "candles_available": candles_count,
   879→            "trie_patterns": all_tries.get("n3").pattern_count if all_tries.get("n3") else 0,
   880→        }
   881→
   882→    except Exception as e:
   883→        return {"ok": False, "error": str(e)}

   952→    Each token gets its own asyncio task that:

   954→      2. Auto-ingests data (if < 500 candles)

   974→            if sym.endswith("USDT"):

   976→            else:
   977→                sym = sym + "/USDT"
   978→

   980→
   981→        # Skip if already running
   982→        existing = _multi_sessions.get(node_id)
   983→        if existing and not existing.get("task").done():
   984→            launched.append({"node_id": node_id, "symbol": sym, "status": "ALREADY_RUNNING"})
   985→            continue
   986→
   987→        # Per-token capital (even split)
   988→        per_capital = req.capital / max(len(tokens), 1)
   989→
   990→        config = LiveConfig(
   991→            symbol=sym,
   992→            timeframe=tf,
   993→            initial_capital=per_capital,
   994→            exchange=exch,
   995→            dry_run=True,
   996→            testnet=False,
   997→            leverage=req.leverage,
   998→            auto_mode=req.auto_mode,
   999→            use_token_profile=True,
  1000→            auto_calibrate=True,
  1001→            regime_aware=True,
  1002→            use_multi_level=True,
  1003→            living_trie=True,
  1004→        )
  1005→
  1006→        session_state = {
  1007→            "node_id": node_id,
  1008→            "symbol": sym,
  1009→            "timeframe": tf,
  1010→            "exchange": exch,
  1011→            "started_at": time.time(),
  1012→            "status": "STARTING",
  1013→            "last_price": 0.0,
  1014→            "pnl_pct": 0.0,
  1015→            "signals": 0,
  1016→            "trades": 0,
  1017→            "candles_processed": 0,
  1018→            "error": "",

  1020→            "regime": "",
  1021→            "pattern_buffer": [],
  1022→            "entropy": 0.0,
  1023→            "websocket_status": "disconnected",
  1024→            "is_running": False,
  1025→            "portfolio_value": 0.0,
  1026→            "win_rate": 0.0,
  1027→            "exposure_pct": 0.0,
  1028→            "validation_verdict": "",

  1030→            "initial_capital": per_capital,
  1031→            "last_update_ts": 0.0,
  1032→            # v0.39.0: track the currently-open position (None when flat) so
  1033→            # /api/multi-status can expose it to the dashboard chart for an
  1034→            # entry-price price-line overlay. Updated by cfg.on_position below.
  1035→            "open_position": None,
  1036→            # v0.39.2: per-session signals ring buffer (cap 50). Populated by
  1037→            # cfg.on_signal below. Previously signals were NEVER shown in
  1038→            # multi-token mode because (a) the v0.39.1 cross-contamination
  1039→            # fix skipped the global _terminal_state singleton in multi-token
  1040→            # mode, and (b) _state_cb ignored the `signal=` kwarg that the
  1041→            # engine forwards via _update_terminal_state(signal={...}). As a
  1042→            # result the dashboard's Signals panel always showed "No signals"
  1043→            # even when the bot was actively generating them, which the user
  1044→            # perceived as "bot not operating". This buffer + the
  1045→            # /api/multi-status + WS-broadcast enrichment (see /ws below)
  1046→            # restore visibility without re-introducing cross-contamination.
  1047→            "signals_history": [],
  1048→        }
  1049→
  1050→        # v0.39.0: Wire on_position callback. The RealtimeTrader fires this
  1051→        # when a position opens or closes (live mode only). We use it to
  1052→        # update _multi_sessions[node_id]["open_position"] in real time so
  1053→        # /api/multi-status can return the entry price to the chart, which
  1054→        # renders a horizontal price line at the entry for the active token.
  1055→        # v0.39.6: On close, we also broadcast an out-of-band WS event
  1056→        # `{"type": "trade_event", "event": "trade_closed", "payload": {...}}`
  1057→        # so the frontend can immediately refresh the "Recently Closed" list
  1058→        # + Hero KPIs without waiting for the next 3s poll.
  1059→        def _on_position_hook(payload, _nid=node_id):
  1060→            try:
  1061→                sess_ref = _multi_sessions.get(_nid)
  1062→                if sess_ref is None:
  1063→                    return
  1064→                if payload.get("action") == "open":

  1197→        async def _run_one_token(_sym=sym, _tf=tf, _exch=exch, _cfg=config, _nid=node_id):
  1198→            sess = _multi_sessions[_nid]
  1199→            try:
  1200→                # Auto-validate first (non-blocking gate)
  1201→                # v0.38.6: Paper trading (dry_run=True) NO debe bloquear el arranque del

  1203→                # usuario pueda VER el sistema operando (con señales+trades) aunque los

  1214→                # wastes 10-30s per token AND creates a duplicate DB row.

  1218→                _needs_validation = (

  1225→                        symbol=_sym, timeframe=_tf, exchange=_exch, capital=_cfg.initial_capital,

  1247→                    sess["validation_verdict"] = _existing_verdict

  1249→                        sess["status"] = "VALIDATION_FAILED"
  1250→                        sess["error"] = "Validation: INSUFFICIENT_DATA (real-money gate)"

  1252→
  1253→                # v0.36.2: Per-session state callback — bridges trader updates to
  1254→                # this session's dict so /api/multi-status returns real values
  1255→                # instead of all zeros. Also drives granular status transitions.

  1284→                        s["entropy"] = float(kwargs["entropy"] or 0)

  1330→                    # and STARTING_TRADER would promote, leaving sessions

  1338→                        s["status"] = "CONNECTING"
  1339→                    elif (kwargs.get("websocket_status") == "warming_up"
  1340→                          and s["status"] in ("STARTING_TRADER", "STARTING", "CONNECTING")):
  1341→                        s["status"] = "WARMING_UP"
  1342→
  1343→                sess["status"] = "STARTING_TRADER"
  1344→                trader = RealtimeTrader(config=_cfg, state_callback=_state_cb)
  1345→                # run_live() runs until cancelled

  1410→@app.get("/api/multi-status")
  1411→async def multi_status() -> dict:
  1412→    """v0.36.0: Live status of all multi-token trading sessions.
  1413→
  1414→    v0.36.2: Now also returns regime, pattern_buffer, entropy, websocket_status,
  1415→    portfolio_value, win_rate, exposure_pct, and last_update_ts per session.
  1416→    v0.38.9: Now also returns `realized_pnl` and `realized_pnl_pct` per session
  1417→    (sum of pnl from closed live trades in SQLite). This lets the Trading tab
  1418→    show realized P&L separately from unrealized, addressing the user's concern
  1419→    that "P&L always shows 0.00% even when trades have closed".
  1420→    """
  1421→    # v0.38.9: Bulk-fetch realized P&L per symbol+timeframe to avoid N+1 queries.
  1422→    # Each session looks up its realized pnl from this dict.
  1423→    realized_map: dict[tuple[str, str], float] = {}
  1424→    try:
  1425→        storage = PPMTStorage()
  1426→        cursor = storage._ensure_conn().cursor()
  1427→        cursor.execute(
  1428→            """SELECT symbol, timeframe, COALESCE(SUM(pnl), 0) as total_pnl
  1429→               FROM trades WHERE source = 'live'
  1430→               GROUP BY symbol, timeframe"""
  1431→        )
  1432→        for row in cursor.fetchall():
  1433→            realized_map[(row[0], row[1])] = float(row[2] or 0)
  1434→        storage.close()
  1435→    except Exception as e:
  1436→        logger.warning(f"multi_status: realized_pnl bulk fetch failed: {e}")
  1437→
  1438→    sessions = []
  1439→    now = time.time()
  1440→    for node_id, sess in _multi_sessions.items():
  1441→        is_active = not sess["task"].done()
  1442→        # v0.36.2: Stale detection — if a session claims to be RUNNING but
  1443→        # hasn't updated state in 60s, mark as STALE so the UI can surface it.
  1444→        last_update = sess.get("last_update_ts", 0)
  1445→        status = sess["status"]
  1446→        if is_active and status == "RUNNING" and last_update > 0 and (now - last_update) > 60:
  1447→            status = "STALE"
  1448→        # v0.38.9: Realized P&L from SQLite (sum of pnl for this symbol+tf, live source)
  1449→        sym = sess["symbol"]
  1450→        tf = sess["timeframe"]
  1451→        realized_pnl = realized_map.get((sym, tf), 0.0)
  1452→        initial_cap = float(sess.get("initial_capital", 0) or 0)
  1453→        realized_pnl_pct = (realized_pnl / initial_cap * 100.0) if initial_cap > 0 else 0.0
  1454→        sessions.append({
  1455→            "node_id": node_id,
  1456→            "symbol": sym,
  1457→            "timeframe": tf,
  1458→            "exchange": sess["exchange"],
  1459→            "started_at": sess["started_at"],
  1460→            "status": status if is_active else (
  1461→                "STOPPED" if sess["status"] in ("STARTING_TRADER", "RUNNING", "CONNECTING", "WARMING_UP") else sess["status"]
  1462→            ),
  1463→            "last_price": sess.get("last_price", 0.0),
  1464→            "pnl_pct": sess.get("pnl_pct", 0.0),
  1465→            "signals": sess.get("signals", 0),
  1466→            "trades": sess.get("trades", 0),
  1467→            "candles_processed": sess.get("candles_processed", 0),
  1468→            "error": sess.get("error", ""),
  1469→            "is_active": is_active,
  1470→            "uptime_seconds": (now - sess["started_at"]) if is_active else 0,
  1471→            # v0.36.2: new fields
  1472→            "regime": sess.get("regime", ""),

  1727→    - Profit Factor > 0.8
