    49тЖТ    49тЖТdef parse_mexc_orders(xlsx_path: Path) -> pd.DataFrame:
    50тЖТ    50тЖТ    """Parse MEXC futures order history."""
    51тЖТ    51тЖТ    LOG.info("Loading MEXC orders from %s", xlsx_path.name)
    52тЖТ    52тЖТ    df = pd.read_excel(xlsx_path)
    53тЖТ    53тЖТ    LOG.info("Loaded %d orders", len(df))
    54тЖТ    54тЖТ    
    55тЖТ    55тЖТ    # Parse timestamps (MEXC gives UTC+2, convert to UTC)
    56тЖТ    56тЖТ    df["_ts"] = pd.to_datetime(df["Tiempo(UTC+02:00)"].astype(str)) - pd.Timedelta(hours=2)

    58тЖТ    58тЖТ    
    59тЖТ    59тЖТ    # Parse symbol: "RIVERUSDT" тЖТ "RIVER"
    60тЖТ    60тЖТ    df["_symbol"] = df["Par de Trading de Futuros"].str.strip().str.upper().str.replace("USDT", "", regex=False).str.strip()
    61тЖТ    61тЖТ    
    62тЖТ    62тЖТ    # Parse direction: MEXC semantics
    63тЖТ    63тЖТ    #   buy long  = OPEN LONG   (PNL always 0)
    64тЖТ    64тЖТ    #   sell long = CLOSE LONG  (PNL = trade result)
    65тЖТ    65тЖТ    #   sell short = OPEN SHORT  (PNL always 0)
    66тЖТ    66тЖТ    #   buy short = CLOSE SHORT  (PNL = trade result)
    67тЖТ    67тЖТ    df["_action"] = df["Direcci├│n"].str.strip().str.lower().map({
    68тЖТ    68тЖТ        "buy long": "open_long",
    69тЖТ    69тЖТ        "sell long": "close_long", 
    70тЖТ    70тЖТ        "sell short": "open_short",
    71тЖТ    71тЖТ        "buy short": "close_short",
    72тЖТ    72тЖТ    })
    73тЖТ    73тЖТ    
    74тЖТ    74тЖТ    # Numeric columns

    91тЖТ    91тЖТdef match_trades(orders: pd.DataFrame) -> pd.DataFrame:
    92тЖТ    92тЖТ    """Match open/close orders into trades (FIFO matching)."""
    93тЖТ    93тЖТ    LOG.info("Matching open/close orders into trades...")
    94тЖТ    94тЖТ    
    95тЖТ    95тЖТ    trades = []
    96тЖТ    96тЖТ    opens = defaultdict(lambda: defaultdict(list))
    97тЖТ    97тЖТ    
    98тЖТ    98тЖТ    for idx, row in orders.iterrows():
    99тЖТ    99тЖТ        action = row.get("_action")
   100тЖТ   100тЖТ        if not action or pd.isna(action):
   101тЖТ   101тЖТ            continue
   102тЖТ   102тЖТ        
   103тЖТ   103тЖТ        symbol = row["_symbol"]
   104тЖТ   104тЖТ        ts = row["_ts"]
   105тЖТ   105тЖТ        price = row["_price"]
   106тЖТ   106тЖТ        pnl = row["_pnl"]
   107тЖТ   107тЖТ        fee = row["_fee"]
   108тЖТ   108тЖТ        qty = row["_qty"]
   109тЖТ   109тЖТ        leverage = row["_leverage"]
   110тЖТ   110тЖТ        
   111тЖТ   111тЖТ        if "open" in action:
   112тЖТ   112тЖТ            direction = "long" if "long" in action else "short"
   113тЖТ   113тЖТ            opens[symbol][direction].append({
   114тЖТ   114тЖТ                "ts": ts, "price": price, "fee": fee, "qty": qty, 
   115тЖТ   115тЖТ                "leverage": leverage, "idx": idx
   116тЖТ   116тЖТ            })
   117тЖТ   117тЖТ        
   118тЖТ   118тЖТ        elif "close" in action:
   119тЖТ   119тЖТ            direction = "long" if "long" in action else "short"
   120тЖТ   120тЖТ            open_list = opens[symbol][direction]
   121тЖТ   121тЖТ            
   122тЖТ   122тЖТ            if not open_list:
   123тЖТ   123тЖТ                trades.append({
   124тЖТ   124тЖТ                    "symbol": symbol, "direction": direction,
   125тЖТ   125тЖТ                    "entry_time": ts - timedelta(minutes=30),
   126тЖТ   126тЖТ                    "entry_price": price, "exit_time": ts, "exit_price": price,
   127тЖТ   127тЖТ                    "pnl": pnl, "fee": fee, "n_entries": 1, "leverage": leverage,
   128тЖТ   128тЖТ                })
   129тЖТ   129тЖТ                continue
   130тЖТ   130тЖТ            
   131тЖТ   131тЖТ            # FIFO matching
   132тЖТ   132тЖТ            matched = []
   133тЖТ   133тЖТ            remaining = qty
   134тЖТ   134тЖТ            while open_list and remaining > 0:
   135тЖТ   135тЖТ                o = open_list.pop(0)
   136тЖТ   136тЖТ                matched.append(o)
   137тЖТ   137тЖТ                remaining -= o["qty"]
   138тЖТ   138тЖТ            
   139тЖТ   139тЖТ            if remaining < 0 and matched:
   140тЖТ   140тЖТ                partial = matched.pop()
   141тЖТ   141тЖТ                partial["qty"] = -remaining
   142тЖТ   142тЖТ                open_list.insert(0, partial)
   143тЖТ   143тЖТ            
   144тЖТ   144тЖТ            if matched:
   145тЖТ   145тЖТ                first = matched[0]
   146тЖТ   146тЖТ                total_fee = sum(o["fee"] for o in matched) + fee
   147тЖТ   147тЖТ                avg_entry = np.average([o["price"] for o in matched],
   148тЖТ   148тЖТ                                       weights=[o["qty"] for o in matched])
   149тЖТ   149тЖТ                trades.append({
   150тЖТ   150тЖТ                    "symbol": symbol, "direction": direction,
   151тЖТ   151тЖТ                    "entry_time": first["ts"], "entry_price": avg_entry,
   152тЖТ   152тЖТ                    "exit_time": ts, "exit_price": price,
   153тЖТ   153тЖТ                    "pnl": pnl, "fee": total_fee,
   154тЖТ   154тЖТ                    "n_entries": len(matched), "leverage": first.get("leverage", leverage),
   155тЖТ   155тЖТ                })
   156тЖТ   156тЖТ    
   157тЖТ   157тЖТ    # Remaining open positions
   158тЖТ   158тЖТ    for sym, dirs in opens.items():
   159тЖТ   159тЖТ        for direction, open_list in dirs.items():
   160тЖТ   160тЖТ            for o in open_list:
   161тЖТ   161тЖТ                trades.append({
   162тЖТ   162тЖТ                    "symbol": sym, "direction": direction,
   163тЖТ   163тЖТ                    "entry_time": o["ts"], "entry_price": o["price"],
   164тЖТ   164тЖТ                    "exit_time": pd.NaT, "exit_price": 0,
   165тЖТ   165тЖТ                    "pnl": 0, "fee": o["fee"],
   166тЖТ   166тЖТ                    "n_entries": 1, "leverage": o.get("leverage", 1),
   167тЖТ   167тЖТ                })
   168тЖТ   168тЖТ    
   169тЖТ   169тЖТ    result = pd.DataFrame(trades)
   170тЖТ   170тЖТ    closed = result[result["pnl"] != 0].copy()
   171тЖТ   171тЖТ    LOG.info("Matched %d total, %d closed (with PnL)", len(result), len(closed))
   172тЖТ   172тЖТ    if len(closed) > 0:
   173тЖТ   173тЖТ        LOG.info("Closed: WR=%.1f%% PnL=%.2f", (closed["pnl"] > 0).mean()*100, closed["pnl"].sum())
   174тЖТ   174тЖТ    return result

   181тЖТ   181тЖТdef download_ohlcv(symbol: str, timeframe: str, start_ts: int, end_ts: int) -> pd.DataFrame:
   182тЖТ   182тЖТ    """Download with disk caching."""
   183тЖТ   183тЖТ    import ccxt
   184тЖТ   184тЖТ    
   185тЖТ   185тЖТ    cache_file = CACHE_DIR / f"{symbol}_{timeframe}.parquet"
   186тЖТ   186тЖТ    
   187тЖТ   187тЖТ    if cache_file.exists():
   188тЖТ   188тЖТ        cached = pd.read_parquet(cache_file)
   189тЖТ   189тЖТ        if len(cached) > 0 and cached["timestamp"].min() <= start_ts + 86400000 and cached["timestamp"].max() >= end_ts - 86400000:
   190тЖТ   190тЖТ            return cached[(cached["timestamp"] >= start_ts) & (cached["timestamp"] <= end_ts)].copy()
   191тЖТ   191тЖТ    
   192тЖТ   192тЖТ    for exchange_id in ["binance", "mexc"]:
   193тЖТ   193тЖТ        try:
   194тЖТ   194тЖТ            exchange = getattr(ccxt, exchange_id)({"enableRateLimit": True})
   195тЖТ   195тЖТ            ccxt_sym = f"{symbol}/USDT"
   196тЖТ   196тЖТ            exchange.load_markets()
   197тЖТ   197тЖТ            if ccxt_sym not in exchange.markets:
   198тЖТ   198тЖТ                continue
   199тЖТ   199тЖТ            
   200тЖТ   200тЖТ            all_ohlcv = []
   201тЖТ   201тЖТ            since = start_ts
   202тЖТ   202тЖТ            while True:
   203тЖТ   203тЖТ                ohlcv = exchange.fetch_ohlcv(ccxt_sym, timeframe, since=since, limit=1000)
   204тЖТ   204тЖТ                if not ohlcv:
   205тЖТ   205тЖТ                    break
   206тЖТ   206тЖТ                all_ohlcv.extend(ohlcv)
   207тЖТ   207тЖТ                last_ts = ohlcv[-1][0]
   208тЖТ   208тЖТ                if last_ts >= end_ts or len(ohlcv) < 1000:
   209тЖТ   209тЖТ                    break
   210тЖТ   210тЖТ                since = last_ts + 1
   211тЖТ   211тЖТ                time.sleep(exchange.rateLimit / 1000)
   212тЖТ   212тЖТ            
   213тЖТ   213тЖТ            if all_ohlcv:
   214тЖТ   214тЖТ                df = pd.DataFrame(all_ohlcv, columns=["timestamp", "open", "high", "low", "close", "volume"])
   215тЖТ   215тЖТ                df = df.drop_duplicates(subset=["timestamp"]).sort_values("timestamp").reset_index(drop=True)
   216тЖТ   216тЖТ                df = df[(df["timestamp"] >= start_ts) & (df["timestamp"] <= end_ts)]
   217тЖТ   217тЖТ                df.to_parquet(cache_file, index=False)
   218тЖТ   218тЖТ                LOG.info("  %s: %d bars from %s (cached)", symbol, len(df), exchange_id)
   219тЖТ   219тЖТ                return df
   220тЖТ   220тЖТ        except Exception as e:
   221тЖТ   221тЖТ            LOG.warning("  %s on %s: %s", symbol, exchange_id, str(e)[:80])
   222тЖТ   222тЖТ            time.sleep(2)
   223тЖТ   223тЖТ    
   224тЖТ   224тЖТ    LOG.warning("  %s: no data available", symbol)

   232тЖТ   232тЖТdef compute_pattern_features(closes, highs, lows, opens, volumes, i, lookback=50):
   233тЖТ   233тЖТ    """Compute ~30 pattern detector features at bar i (only past data)."""
   234тЖТ   234тЖТ    n = len(closes)
   235тЖТ   235тЖТ    if i < lookback or i >= n:
   236тЖТ   236тЖТ        return None
   237тЖТ   237тЖТ    
   238тЖТ   238тЖТ    c = closes[:i+1]
   239тЖТ   239тЖТ    h = highs[:i+1]
   240тЖТ   240тЖТ    l = lows[:i+1]
   241тЖТ   241тЖТ    o = opens[:i+1]
   242тЖТ   242тЖТ    v = volumes[:i+1]
   243тЖТ   243тЖТ    f = {}
   244тЖТ   244тЖТ    
   245тЖТ   245тЖТ    # тФАтФА BREAKOUT тФАтФА
   246тЖТ   246тЖТ    high_20 = np.max(h[-20:])
   247тЖТ   247тЖТ    low_20 = np.min(l[-20:])
   248тЖТ   248тЖТ    range_20 = high_20 - low_20
   249тЖТ   249тЖТ    if range_20 > 0:
   250тЖТ   250тЖТ        f["close_position_20"] = (c[-1] - low_20) / range_20
   251тЖТ   251тЖТ        f["is_breakout_up"] = float(c[-1] >= high_20)
   252тЖТ   252тЖТ        f["is_breakout_down"] = float(c[-1] <= low_20)
   253тЖТ   253тЖТ        f["breakout_strength"] = max(0, abs(c[-1] - (high_20 if c[-1] >= high_20 else low_20)) / range_20)
   254тЖТ   254тЖТ    else:
   255тЖТ   255тЖТ        f["close_position_20"] = 0.5; f["is_breakout_up"] = 0; f["is_breakout_down"] = 0; f["breakout_strength"] = 0
   256тЖТ   256тЖТ    
   257тЖТ   257тЖТ    # тФАтФА EMA BOUNCE тФАтФА
   258тЖТ   258тЖТ    ema9 = _ema(c, 9); ema21 = _ema(c, 21); ema50 = _ema(c, 50) if len(c) >= 50 else _ema(c, min(len(c), 21))

   265тЖТ   265тЖТ    else:
   266тЖТ   266тЖТ        f["dist_ema9_atr"] = 0; f["dist_ema21_atr"] = 0; f["dist_ema50_atr"] = 0
   267тЖТ   267тЖТ    
   268тЖТ   268тЖТ    f["ema21_bounce"] = 0.0; f["ema50_bounce"] = 0.0
   269тЖТ   269тЖТ    if atr > 0 and len(c) >= 6:
   270тЖТ   270тЖТ        for j in range(1, min(4, len(c))):
   271тЖТ   271тЖТ            if l[-j] <= ema21[-j] + 0.3*atr and h[-j] >= ema21[-j] - 0.3*atr:
   272тЖТ   272тЖТ                f["ema21_bounce"] = 1.0; break
   273тЖТ   273тЖТ        for j in range(1, min(6, len(c))):
   274тЖТ   274тЖТ            if l[-j] <= ema50[-j] + 0.3*atr and h[-j] >= ema50[-j] - 0.3*atr:
   275тЖТ   275тЖТ                f["ema50_bounce"] = 1.0; break

   279тЖТ   279тЖТ

   290тЖТ   290тЖТ                if s > 0: bb_list.append(4*sd/s)
   291тЖТ   291тЖТ        f["squeeze_score"] = 1.0 - np.mean([w <= bb_width for w in bb_list]) if bb_list else 0.5
   292тЖТ   292тЖТ    else:
   293тЖТ   293тЖТ        f["bb_width"] = 0; f["squeeze_score"] = 0.5
   294тЖТ   294тЖТ    
   295тЖТ   295тЖТ    # тФАтФА VOLUME тФАтФА
   296тЖТ   296тЖТ    if len(v) >= 20:
   297тЖТ   297тЖТ        vol_ma = np.mean(v[-20:])
   298тЖТ   298тЖТ        f["vol_ratio"] = v[-1] / max(vol_ma, 1e-10)
   299тЖТ   299тЖТ        f["vol_ratio_3"] = np.mean(v[-3:]) / max(vol_ma, 1e-10)

   306тЖТ   306тЖТ    f["body_ratio"] = abs(body) / bar_range

   315тЖТ   315тЖТ    if len(c) >= 2:
   316тЖТ   316тЖТ        prev_body = c[-2] - o[-2]

   318тЖТ   318тЖТ        f["is_bearish_engulf"] = float(body < 0 and prev_body > 0 and c[-1] < o[-2] and o[-1] > c[-2])

   329тЖТ   329тЖТ            f["is_pullback_long"] = float(f["pullback_depth"] > 0.3 and trend_dir > 0)
   330тЖТ   330тЖТ            f["is_pullback_short"] = 0.0
   331тЖТ   331тЖТ        elif trend_dir < 0:
   332тЖТ   332тЖТ            f["pullback_depth"] = (c[-1] - low_6) / (high_6 - low_6)
   333тЖТ   333тЖТ            f["is_pullback_short"] = float(f["pullback_depth"] > 0.3 and trend_dir < 0)
   334тЖТ   334тЖТ            f["is_pullback_long"] = 0.0
   335тЖТ   335тЖТ        else:
   336тЖТ   336тЖТ            f["pullback_depth"] = 0.5; f["is_pullback_long"] = 0; f["is_pullback_short"] = 0
   337тЖТ   337тЖТ    else:

   345тЖТ   345тЖТ    if len(c) > 10:
   346тЖТ   346тЖТ        rets = np.diff(c[-50:]) / np.maximum(c[-50:-1], 1e-10) if len(c) > 50 else np.diff(c) / np.maximum(c[:-1], 1e-10)
   347тЖТ   347тЖТ        f["momentum_z"] = (f["ret_1_pct"]/100) / max(np.std(rets), 1e-10) if len(rets) > 5 else 0
   348тЖТ   348тЖТ    else:
   349тЖТ   349тЖТ        f["momentum_z"] = 0
   350тЖТ   350тЖТ

   352тЖТ   352тЖТ    for j in range(1, min(10, len(c)-1)):

   361тЖТ   361тЖТ    f["near_low_50"] = float(abs(c[-1] - low_50) / max(low_50, 1e-10) < 0.01)
   362тЖТ   362тЖТ    
   363тЖТ   363тЖТ    # тФАтФА RANGE EXPANSION тФАтФА
   364тЖТ   364тЖТ    if len(h) >= 3:
   365тЖТ   365тЖТ        range_3 = np.mean([h[-j] - l[-j] for j in range(1, min(4, len(h)+1))])

   368тЖТ   368тЖТ    else:
   369тЖТ   369тЖТ        f["range_expansion"] = 1.0
   370тЖТ   370тЖТ    
   371тЖТ   371тЖТ    # тФАтФА V-REVERSAL тФАтФА
   372тЖТ   372тЖТ    if len(c) >= 7:
   373тЖТ   373тЖТ        r1 = (c[-4] - c[-7]) / max(c[-7], 1e-10)
   374тЖТ   374тЖТ        r2 = (c[-1] - c[-4]) / max(c[-4], 1e-10)
   375тЖТ   375тЖТ        f["v_reversal"] = float(np.sign(r1) != np.sign(r2) and abs(r2) > abs(r1)*0.5)
   376тЖТ   376тЖТ    else:
   377тЖТ   377тЖТ        f["v_reversal"] = 0.0
   378тЖТ   378тЖТ

   384тЖТ   384тЖТdef _ema(data, period):
   385тЖТ   385тЖТ    if len(data) < period: return np.full(len(data), data[-1] if len(data) > 0 else 0)
   386тЖТ   386тЖТ    alpha = 2 / (period + 1)
   387тЖТ   387тЖТ    result = np.zeros(len(data)); result[0] = data[0]
   388тЖТ   388тЖТ    for i in range(1, len(data)): result[i] = alpha * data[i] + (1 - alpha) * result[i-1]
   389тЖТ   389тЖТ    return result

   391тЖТ   391тЖТdef _atr(highs, lows, closes, period=14):
   392тЖТ   392тЖТ    if len(closes) < 2: return 0.0
   393тЖТ   393тЖТ    n = min(period, len(closes) - 1)
   394тЖТ   394тЖТ    trs = []
   395тЖТ   395тЖТ    for i in range(max(1, len(closes) - n), len(closes)):
   396тЖТ   396тЖТ        tr = max(highs[i] - lows[i], abs(highs[i] - closes[i-1]), abs(lows[i] - closes[i-1]))
   397тЖТ   397тЖТ        trs.append(tr)
   398тЖТ   398тЖТ    return np.mean(trs) if trs else 0.0
